function repairEffectEvaluationGoogleForm() {
  ensureV41Structure_();

  const form = buildEffectEvaluationForm_();
  prepareEffectEvaluationForm_(form);

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  form.setDestination(FormApp.DestinationType.SPREADSHEET, ss.getId());

  Utilities.sleep(1000);

  const info = saveEffectEvaluationFormSettings_(form);
  installEffectEvaluationSubmitTrigger_();

  SpreadsheetApp.getUi().alert(
    '효과평가 Google Form 복구 완료',
    '기존 연결 대신 정상 문항이 포함된 새 Form을 만들고 연결했습니다.\n\n' +
    '응답용 주소: ' + info.formUrl + '\n\n' +
    '편집용 주소: ' + info.editUrl + '\n\n' +
    '이제 교사용 웹앱을 새로고침한 뒤 학생을 다시 선택하세요.',
    SpreadsheetApp.getUi().ButtonSet.OK
  );

  return info;
}

function ensureEffectEvaluationForm_() {
  ensureV41Structure_();
  const settings = getSettings_();
  let form = null;
  const savedId = String(settings[EFFECT_FORM_SETTING_KEYS.ID] || '').trim();

  if (savedId) {
    try {
      form = FormApp.openById(savedId);
    } catch (error) {
      form = null;
    }
  }

  // 저장된 Form이 비어 있거나 문항명이 달라진 경우 새 Form으로 자동 복구합니다.
  if (!form || !isEffectEvaluationFormValid_(form)) {
    form = buildEffectEvaluationForm_();
  }

  prepareEffectEvaluationForm_(form);

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  if (
    form.getDestinationType() !== FormApp.DestinationType.SPREADSHEET ||
    String(form.getDestinationId() || '') !== String(ss.getId())
  ) {
    form.setDestination(FormApp.DestinationType.SPREADSHEET, ss.getId());
  }

  Utilities.sleep(500);
  return { form, ...saveEffectEvaluationFormSettings_(form) };
}

function prepareEffectEvaluationForm_(form) {
  // 최신 Google Forms의 게시/응답자 권한 방식과 기존 방식을 모두 지원합니다.
  try {
    if (
      typeof form.supportsAdvancedResponderPermissions === 'function' &&
      form.supportsAdvancedResponderPermissions()
    ) {
      form.setPublished(true);
    } else {
      form.setAcceptingResponses(true);
    }
  } catch (error) {
    // 일부 Workspace 환경에서는 게시 API가 지원되지 않을 수 있습니다.
    try {
      form.setAcceptingResponses(true);
    } catch (ignored) {}
  }

  try {
    form.setCollectEmail(false);
    form.setLimitOneResponsePerUser(false);
  } catch (ignored) {}
}

function saveEffectEvaluationFormSettings_(form) {
  const info = {
    formId: form.getId(),
    formUrl: form.getPublishedUrl(),
    editUrl: form.getEditUrl()
  };

  setSetting_(EFFECT_FORM_SETTING_KEYS.ID, info.formId);
  setSetting_(EFFECT_FORM_SETTING_KEYS.URL, info.formUrl);
  setSetting_(EFFECT_FORM_SETTING_KEYS.EDIT_URL, info.editUrl);
  return info;
}

function isEffectEvaluationFormValid_(form) {
  try {
    const items = form.getItems();
    const titles = new Set(
      items.map(item => String(item.getTitle() || '').trim()).filter(Boolean)
    );

    const requiredTitles = [
      EFFECT_FORM_TITLES.CASE_ID,
      EFFECT_FORM_TITLES.STUDENT_NAME,
      EFFECT_FORM_TITLES.GRADE,
      EFFECT_FORM_TITLES.CLASS_NO,
      EFFECT_FORM_TITLES.POINT,
      EFFECT_FORM_TITLES.PRE.TARGET,
      EFFECT_FORM_TITLES.POST.TARGET
    ];

    const answerableTypes = [
      FormApp.ItemType.TEXT,
      FormApp.ItemType.PARAGRAPH_TEXT,
      FormApp.ItemType.MULTIPLE_CHOICE,
      FormApp.ItemType.CHECKBOX,
      FormApp.ItemType.SCALE
    ];

    const answerableCount = items.filter(item =>
      answerableTypes.includes(item.getType())
    ).length;

    return (
      answerableCount >= 15 &&
      requiredTitles.every(title => titles.has(title))
    );
  } catch (error) {
    return false;
  }
}

  ]);

}

function installEffectEvaluationSubmitTrigger_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const matches = ScriptApp.getProjectTriggers()
    .filter(trigger => trigger.getHandlerFunction() === 'handleEffectEvaluationFormSubmit');

  if (!matches.length) {
    ScriptApp.newTrigger('handleEffectEvaluationFormSubmit')
      .forSpreadsheet(ss)
      .onFormSubmit()
      .create();
    return;
  }

  // 중복 트리거가 생긴 경우 첫 번째만 유지합니다.
  matches.slice(1).forEach(trigger => ScriptApp.deleteTrigger(trigger));
}

function handleEffectEvaluationFormSubmit(e) {
  try {
    if (!e || !e.namedValues) return;
    const named = e.namedValues;
    const caseId = formNamedValue_(named, EFFECT_FORM_TITLES.CASE_ID);
    const point = formNamedValue_(named, EFFECT_FORM_TITLES.POINT);
    if (!caseId || !['사전', '사후'].includes(point)) return;

    const detail = getCaseDetailRaw_(caseId);
    const student = detail.student || {};
    const submittedName = formNamedValue_(named, EFFECT_FORM_TITLES.STUDENT_NAME);
    const submittedGrade = formNamedValue_(named, EFFECT_FORM_TITLES.GRADE);
    const submittedClass = formNamedValue_(named, EFFECT_FORM_TITLES.CLASS_NO);
    const titles = point === '사후' ? EFFECT_FORM_TITLES.POST : EFFECT_FORM_TITLES.PRE;
    const warnings = [];

    if (submittedName && normalize_(submittedName) !== normalize_(student['학생명'])) {
      warnings.push('제출 학생명과 상담 케이스 학생명이 다름');
    }
    if (submittedGrade && Number(parseNumber_(submittedGrade)) !== Number(student['학년'])) {
      warnings.push('제출 학년과 상담 케이스 학년이 다름');
    }
    if (submittedClass && Number(parseNumber_(submittedClass)) !== Number(student['반'])) {
      warnings.push('제출 반과 상담 케이스 반이 다름');
    }

    const submittedAt = e.values && e.values.length ? safeDateForSheet_(e.values[0], new Date()) : new Date();
    const support = formNamedValue_(named, titles.SUPPORT);
    const barrier = formNamedValue_(named, titles.BARRIER);
    const postHelpful = point === '사후'
      ? formNamedValue_(named, EFFECT_FORM_TITLES.POST.HELPFUL)
      : '';

    saveEffectEvaluation({
      caseId,
      evaluationPoint: point,
      evaluationDate: submittedAt,
      dietSatisfaction: formNamedValue_(named, titles.DIET),
      goalUnderstanding: formNamedValue_(named, titles.GOAL_UNDERSTANDING),
      confidence: formNamedValue_(named, titles.CONFIDENCE),
      importance: formNamedValue_(named, titles.IMPORTANCE),
      readiness: formNamedValue_(named, titles.READINESS),
      targetBehavior: formNamedValue_(named, titles.TARGET),
      behaviorCount: formNamedValue_(named, titles.COUNT),
      methodUnderstanding: formNamedValue_(named, titles.METHOD),
      helpSeeking: formNamedValue_(named, titles.HELP),
      counselingSatisfaction: point === '사후'
        ? formNamedValue_(named, EFFECT_FORM_TITLES.POST.SATISFACTION)
        : '',
      counselingHelpfulness: point === '사후'
        ? formNamedValue_(named, EFFECT_FORM_TITLES.POST.COUNSELING_HELP)
        : '',
      helpfulPoint: postHelpful || support,
      difficulty: barrier,
      maintenanceBehavior: point === '사후'
        ? formNamedValue_(named, EFFECT_FORM_TITLES.POST.MAINTAIN)
        : '',
      changeAfterCounseling: point === '사후'
        ? formNamedValue_(named, EFFECT_FORM_TITLES.POST.CHANGE)
        : '',
      needMoreCounseling: point === '사후'
        ? formNamedValue_(named, EFFECT_FORM_TITLES.POST.NEED_MORE)
        : '',
      teacherMessage: point === '사후'
        ? formNamedValue_(named, EFFECT_FORM_TITLES.POST.MESSAGE)
        : '',
      supportFactors: support,
      barrierFactors: barrier,
      responseSource: 'Google Form 자동연계',
      formSubmittedAt: submittedAt,
      note: warnings.join(' / ')
    });
  } catch (error) {
    try {
      log_('ERROR', SHEETS.EVALS, 'FORM_SUBMIT', '효과평가 Form 자동연계 실패: ' + error.message);
    } catch (ignore) {}
    throw error;
  }
}

function formNamedValue_(namedValues, title) {
  const value = namedValues[title];
  if (Array.isArray(value)) return value.join(', ').trim();
  return String(value || '').trim();
}

function getEffectEvaluationFormLinks(caseId) {
  validateRequired_({ caseId }, ['caseId']);
  const info = ensureEffectEvaluationForm_();
  installEffectEvaluationSubmitTrigger_();

  const detail = getCaseDetailRaw_(caseId);
  const goals = (detail.goals || []).slice().sort((a, b) => {
    const ad = asDate_(a['시작일']);
    const bd = asDate_(b['시작일']);
    return (ad ? ad.getTime() : 0) - (bd ? bd.getTime() : 0);
  });
  const latestGoal = goals.length ? goals[goals.length - 1] : {};
  const targetBehavior = latestGoal['목표문장'] || detail.case['주상담주제'] || '';

  return toClient_({
    formUrl: info.formUrl,
    editUrl: info.editUrl,
    preUrl: buildEffectEvaluationPrefilledUrl_(
      info.form,
      detail,
      '사전',
      targetBehavior
    ),
    postUrl: buildEffectEvaluationPrefilledUrl_(
      info.form,
      detail,
      '사후',
      targetBehavior
    ),
    studentName: detail.student['학생명'] || '',
    gradeClass: formatGradeClass_(detail.student),
    targetBehavior
  });
}

function buildEffectEvaluationPrefilledUrl_(form, detail, point, targetBehavior) {
  if (!isEffectEvaluationFormValid_(form)) {
    throw new Error(
      '효과평가 설문 문항이 비어 있거나 변경되었습니다. ' +
      '스프레드시트 메뉴에서 “V4.2 효과평가 Form 오류복구·재생성”을 실행하세요.'
    );
  }

  prepareEffectEvaluationForm_(form);

  let response = form.createResponse();
  let responseCount = 0;
  const itemMap = {};

  form.getItems().forEach(item => {
    const title = String(item.getTitle() || '').trim();
    if (title) itemMap[title] = item;
  });

  const add = (title, value) => {
    if (value === '' || value === null || value === undefined) return;
    const item = itemMap[title];
    if (!item) return;

    const text = String(value);
    let itemResponse = null;

    switch (item.getType()) {
      case FormApp.ItemType.TEXT:
        itemResponse = item.asTextItem().createResponse(text);
        break;
      case FormApp.ItemType.MULTIPLE_CHOICE:
        itemResponse = item.asMultipleChoiceItem().createResponse(text);
        break;
      default:
        return;
    }

    response = response.withItemResponse(itemResponse);
    responseCount += 1;
  };

  add(EFFECT_FORM_TITLES.CASE_ID, detail.case['케이스번호']);
  add(EFFECT_FORM_TITLES.STUDENT_NAME, detail.student['학생명'] || '');
  add(EFFECT_FORM_TITLES.GRADE, detail.student['학년'] || '');
  add(EFFECT_FORM_TITLES.CLASS_NO, detail.student['반'] || '');
  add(EFFECT_FORM_TITLES.POINT, point);
  add(
    point === '사후'
      ? EFFECT_FORM_TITLES.POST.TARGET
      : EFFECT_FORM_TITLES.PRE.TARGET,
    targetBehavior
  );

  if (responseCount === 0) {
    throw new Error(
      '설문 문항과 자동입력 항목을 연결하지 못했습니다. ' +
      '“V4.2 효과평가 Form 오류복구·재생성”을 실행하세요.'
    );
  }

  try {
    return response.toPrefilledUrl();
  } catch (error) {
    // Workspace의 Form 게시 상태가 늦게 반영되는 경우 한 번 더 시도합니다.
    prepareEffectEvaluationForm_(form);
    Utilities.sleep(1200);
    return response.toPrefilledUrl();
  }
}

function migrateCaseStatusesV47_() {
  const sheet = getSheet_(SHEETS.CASES);
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return;

  const headers = data[0];
  const statusCol = headers.indexOf('현재단계');
  if (statusCol < 0) return;

  const updates = [];
  for (let r = 1; r < data.length; r++) {
    const current = String(data[r][statusCol] || '').trim();
    if (current === '재상담 필요') {
      updates.push({ row: r + 1, value: '추적상담 예정' });
    }
  }

  updates.forEach(item => sheet.getRange(item.row, statusCol + 1).setValue(item.value));
}

function organizeWorkbookTabsV47_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  let targetPosition = 1;
  V47_SHEET_ORDER.forEach(name => {
    const sheet = ss.getSheetByName(name);
    if (!sheet) return;

    try {
      if (sheet.isSheetHidden()) sheet.showSheet();
    } catch (error) {
      console.log(name + ' 표시 실패: ' + error.message);
    }

    ss.setActiveSheet(sheet);
    ss.moveActiveSheet(targetPosition);
    targetPosition += 1;

    if (V47_TAB_COLORS[name]) {
      sheet.setTabColor(V47_TAB_COLORS[name]);
    }
  });

  ss.getSheets().forEach(sheet => {
    const name = sheet.getName();
    const isRawFormSheet = /^설문지 응답/.test(name) || /^Form Responses/i.test(name);
    const shouldHide = isRawFormSheet || V47_HIDDEN_SHEETS.includes(name);
    if (!shouldHide) return;

    try {
      sheet.hideSheet();
    } catch (error) {
      console.log(name + ' 숨김 실패: ' + error.message);
    }
  });

  const dashboard = ss.getSheetByName('대시보드');
  if (dashboard) ss.setActiveSheet(dashboard);
}

function ensureV36Structure_() {
  ensureSheetHeaders_(SHEETS.EVALS, EFFECT_HEADERS);
  ensureSheetHeaders_(SHEETS.GROWTH, GROWTH_HEADERS);
  ensureSheetHeaders_(SHEETS.DIAG, DIAG_V36_HEADERS);

  const sheet = getSheet_(SHEETS.DIAG);
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return;

  const headers = data[0];
  const idCol = headers.indexOf('진단ID');
  const caseCol = headers.indexOf('케이스번호');
  const noCol = headers.indexOf('검사차수');
  const pointCol = headers.indexOf('평가시점');

  const counters = {};
  for (let r = 1; r < data.length; r++) {
    const caseId = String(data[r][caseCol] || '').trim();
    if (!caseId) continue;

    counters[caseId] = (counters[caseId] || 0) + 1;
    const no = Number(data[r][noCol] || counters[caseId]);

    if (!data[r][idCol]) {
      sheet.getRange(r + 1, idCol + 1).setValue(
        'DG-' + caseId + '-' + Utilities.formatString('%02d', no)
      );
    }
    if (!data[r][noCol]) sheet.getRange(r + 1, noCol + 1).setValue(no);
    if (!data[r][pointCol]) {
      sheet.getRange(r + 1, pointCol + 1).setValue(no === 1 ? '사전' : '추적');
    }
  }
}

function ensureSheetHeaders_(sheetName, requiredHeaders) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.getRange(1, 1, 1, requiredHeaders.length).setValues([requiredHeaders]);
    sheet.setFrozenRows(1);
    return sheet;
  }

  const lastCol = Math.max(sheet.getLastColumn(), 1);
  const currentHeaders = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  let nextCol = currentHeaders.length + 1;

  requiredHeaders.forEach(header => {
    if (!currentHeaders.includes(header)) {
      sheet.getRange(1, nextCol).setValue(header);
      currentHeaders.push(header);
      nextCol += 1;
    }
  });

  sheet.setFrozenRows(1);
  return sheet;
}

  const ui = SpreadsheetApp.getUi();
  const sheet = getSheet_(SHEETS.CONSENTS);
  const data = sheet.getDataRange().getValues();
  const headers = data[0] || [];
  if (headers.includes('동의토큰') && headers.includes('확인상태')) {
    ui.alert('이미 V3 보호자 동의 구조입니다.');
    return;
  }
  const newHeaders = [
    '케이스번호','동의토큰','동의링크','확인상태','보호자명','학생과의관계','보호자연락처',
    '학생참여의사','상담동의','개인정보동의','민감정보동의','진단결과활용동의',
    'AI보조안내확인','발송일','제출일시','동의일','동의서파일URL','확인일','확인자','비고'
  ];
  const oldRows = data.slice(1).filter(row => row.some(v => v !== '' && v !== null));
  const intakeMap = {};
  getSheetObjects_(SHEETS.INTAKE).forEach(r => intakeMap[String(r['접수ID'])] = r);
  const caseMap = {};
  getSheetObjects_(SHEETS.CASES).forEach(r => caseMap[String(r['케이스번호'])] = r);
  const idx = name => headers.indexOf(name);
  const val = (row, name) => idx(name) >= 0 ? row[idx(name)] : '';
  const migrated = oldRows.map(row => {
    const caseId = val(row, '케이스번호');
    const caseRow = caseMap[String(caseId)] || {};
    const intake = intakeMap[String(caseRow['접수ID'])] || {};
    const submitted = val(row, '동의일');
    return [
      caseId,'','',submitted ? '교사 확인 필요' : (val(row,'발송일') ? '동의 요청' : '미발송'),
      '','',intake['연락처'] || '','미확인',
      val(row,'상담동의') || '미확인',val(row,'개인정보동의') || '미확인',val(row,'민감정보동의') || '미확인',
      '미확인',val(row,'AI보조안내확인') || '미확인',val(row,'발송일') || '',submitted || '',submitted || '',
      val(row,'동의서파일URL') || '','',val(row,'확인자') || '',val(row,'비고') || ''
    ];
  });
  sheet.clearContents();
  sheet.getRange(1,1,1,newHeaders.length).setValues([newHeaders]);
  if (migrated.length) sheet.getRange(2,1,migrated.length,newHeaders.length).setValues(migrated);
  sheet.setFrozenRows(1);
  log_('MIGRATE', SHEETS.CONSENTS, 'V2-V3', '보호자동의 시트 구조 업그레이드');
  ui.alert('V3 보호자 동의 구조로 업그레이드했습니다. 기존 동의는 진단결과 활용 동의가 없어 재확인이 필요할 수 있습니다.');
}

/**
 * 케이스번호를 기준으로 학생계정과 최근 연계활동을 읽습니다.
 * 탐험소 연결 오류가 있어도 매니저의 기존 상담화면은 계속 열리도록 합니다.
 */
function getTasteMindExplorerSnapshot_(caseId, skipAutoSync) {
  const config = getTasteMindConfig_();

  try {
    if (!skipAutoSync) {
      syncTasteMindCaseSafe_(caseId, {
        allowCreate: true,
        source: '학생 종합화면 자동 동기화'
      });
    }

    const ss = SpreadsheetApp.openById(config.spreadsheetId);
    const accounts = getExternalSheetObjects_(ss, TASTE_MIND_DEFAULTS.ACCOUNT_SHEET);
    const links = getExternalSheetObjects_(ss, TASTE_MIND_DEFAULTS.LINK_SHEET);
    const meals = getExternalSheetObjects_(ss, TASTE_MIND_DEFAULTS.MEAL_SHEET);

    const account = accounts.find(row =>
      String(row['케이스번호'] || '') === String(caseId)
    );

    if (!account) {
      const eligibility = getTasteMindEligibility_(caseId);
      return {
        connected: true,
        hasAccount: false,
        eligible: eligibility.eligible,
        message: eligibility.message,
        webAppUrl: config.webAppUrl,
        latestActivity: null,
        pendingActivities: [],
        pendingCount: 0,
        completedSessionCount: 0,
        totalSessionCount: 5
      };
    }

    const related = links
      .filter(row => String(row['케이스번호'] || '') === String(caseId))
      .sort((a, b) =>
        tasteMindComparable_(b['전송일시']) -
        tasteMindComparable_(a['전송일시'])
      );

    const pending = related.filter(row =>
      String(row['반영상태'] || '') !== '반영완료'
    );

    const completed = getTasteMindCompletedSessionCount_(caseId, meals);

    return {
      connected: true,
      hasAccount: true,
      eligible: true,
      accountId: account['계정ID'] || '',
      studentCode: account['학생코드'] || '',
      caseId: account['케이스번호'] || caseId,
      alias: account['탐험가별명'] || '',
      explorerCode: account['탐험코드'] || '',
      studentActivityLabel: account['학생표시활동'] || '',
      currentJourney: completed >= 5 ? 5 : completed + 1,
      completedSessionCount: completed,
      totalSessionCount: 5,
      managerStatus: account['매니저현재단계'] || '',
      currentMission: account['최근실천목표'] || '',
      accountStatus: account['계정상태'] || '',
      lastSyncAt: formatTasteMindDateTime_(account['최근동기화']),
      lastAccessAt: formatTasteMindDateTime_(account['최근접속']),
      webAppUrl: config.webAppUrl,
      pendingCount: pending.length,
      pendingActivities: pending.slice(0, 10).map(mapTasteMindLinkRow_),
      latestActivity: related.length ? mapTasteMindLinkRow_(related[0]) : null
    };
  } catch (error) {
    return {
      connected: false,
      hasAccount: false,
      eligible: false,
      message: error.message || String(error),
      webAppUrl: config.webAppUrl,
      latestActivity: null,
      pendingActivities: [],
      pendingCount: 0,
      completedSessionCount: 0,
      totalSessionCount: 5
    };
  }
}

/**
 * 매니저에서 학생 탐험코드를 재발급합니다.
 */
function regenerateTasteMindExplorerCode(caseId) {
  validateRequired_({ caseId }, ['caseId']);

  const config = getTasteMindConfig_();
  const ss = SpreadsheetApp.openById(config.spreadsheetId);
  const sheet = ss.getSheetByName(TASTE_MIND_DEFAULTS.ACCOUNT_SHEET);

  if (!sheet) throw new Error('탐험소 학생계정 시트가 없습니다.');

  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    const values = sheet.getDataRange().getValues();
    if (values.length < 2) {
      throw new Error('탐험소에 등록된 학생계정이 없습니다.');
    }

    const headers = values[0].map(value => String(value || '').trim());
    const caseCol = headers.indexOf('케이스번호');
    const codeCol = headers.indexOf('탐험코드');
    const tokenCol = headers.indexOf('접속토큰');
    const reissueCol = headers.indexOf('코드재발급일');
    const nameCol = headers.indexOf('학생명');

    if (caseCol < 0 || codeCol < 0) {
      throw new Error('학생계정 시트의 열 구조를 확인하세요.');
    }

    let rowNumber = -1;

    for (let i = 1; i < values.length; i += 1) {
      if (String(values[i][caseCol] || '') === String(caseId)) {
        rowNumber = i + 1;
        break;
      }
    }

    if (rowNumber < 0) {
      throw new Error('해당 상담케이스의 탐험소 계정을 찾지 못했습니다.');
    }

    const used = new Set(
      values.slice(1)
        .map(row => String(row[codeCol] || '').trim())
        .filter(Boolean)
    );

    let nextCode = '';

    for (let attempt = 0; attempt < 10000; attempt += 1) {
      const candidate = String(
        Math.floor(1000 + Math.random() * 9000)
      );

      if (!used.has(candidate)) {
        nextCode = candidate;
        break;
      }
    }

    if (!nextCode) {
      throw new Error('새 탐험코드를 생성하지 못했습니다.');
    }

    sheet.getRange(rowNumber, codeCol + 1).setValue(nextCode);

    if (tokenCol >= 0) {
      sheet.getRange(rowNumber, tokenCol + 1)
        .setValue(Utilities.getUuid().replace(/-/g, ''));
    }

    if (reissueCol >= 0) {
      sheet.getRange(rowNumber, reissueCol + 1)
        .setValue(new Date());
    }

    return toClient_({
      ok: true,
      caseId,
      studentName: nameCol >= 0
        ? values[rowNumber - 1][nameCol] || ''
        : '',
      explorerCode: nextCode
    });
  } finally {
    lock.releaseLock();
  }
}

function getTasteMindConfig_() {
  const props = PropertiesService.getScriptProperties();

  return {
    spreadsheetId:
      props.getProperty(TASTE_MIND_PROPERTIES.SPREADSHEET_ID) ||
      TASTE_MIND_DEFAULTS.SPREADSHEET_ID,
    webAppUrl:
      props.getProperty(TASTE_MIND_PROPERTIES.WEB_APP_URL) ||

  };
}

function getExternalSheetObjects_(ss, sheetName) {
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) return [];

  const values = sheet.getDataRange().getValues();
  if (values.length < 2) return [];

  const headers = values[0].map(value =>
    String(value || '').trim()
  );

  return values.slice(1)
    .filter(row => row.some(value =>
      value !== '' && value !== null
    ))
    .map(row => {
      const object = {};

      headers.forEach((header, index) => {
        if (header) object[header] = row[index];
      });

      return object;
    });
}

function tasteMindComparable_(value) {
  if (!value) return 0;

  if (Object.prototype.toString.call(value) === '[object Date]') {
    return isNaN(value.getTime()) ? 0 : value.getTime();
  }

  const date = new Date(value);
  return isNaN(date.getTime()) ? 0 : date.getTime();
}

function formatTasteMindDateTime_(value) {
  if (!value) return '';

  const date = Object.prototype.toString.call(value) === '[object Date]'
    ? value
    : new Date(value);

  if (isNaN(date.getTime())) return String(value || '');

  return Utilities.formatDate(
    date,
    TASTE_MIND_DEFAULTS.TIMEZONE,
    'yyyy-MM-dd HH:mm'
  );
}

function syncTasteMindCase_(caseId, options) {
  options = options || {};

  const caseRow = getSheetObjects_(SHEETS.CASES).find(row =>
    String(row['케이스번호'] || '') === String(caseId)
  );
  if (!caseRow) throw new Error('상담케이스를 찾을 수 없습니다.');

  const student = studentMap_()[caseRow['학생코드']] || {};
  if (!student['학생코드']) throw new Error('학생정보를 찾을 수 없습니다.');

  const consent = getSheetObjects_(SHEETS.CONSENTS).find(row =>
    String(row['케이스번호'] || '') === String(caseId)
  ) || {};

  const eligibility = getTasteMindEligibility_(caseId, caseRow, consent);
  const config = getTasteMindConfig_();
  const ss = SpreadsheetApp.openById(config.spreadsheetId);
  const accountSheet = ss.getSheetByName(TASTE_MIND_DEFAULTS.ACCOUNT_SHEET);
  const missionSheet = ss.getSheetByName(TASTE_MIND_DEFAULTS.MISSION_SHEET);

  if (!accountSheet || !missionSheet) {
    throw new Error('탐험소 학생계정 또는 실천미션 시트가 없습니다.');
  }

  const accounts = getExternalSheetObjects_(ss, TASTE_MIND_DEFAULTS.ACCOUNT_SHEET);
  const existing = accounts.find(row =>
    String(row['케이스번호'] || '') === String(caseId)
  ) || null;

  if (!existing && !eligibility.eligible) {
    return {
      ok: true,
      created: false,
      updated: false,
      eligible: false,
      reason: eligibility.message
    };
  }

  const status = String(caseRow['현재단계'] || '');
  const assent = String(consent['학생참여의사'] || '');
  const accountStatus =
    status === '종결' || assent === '참여하지 않음'
      ? '종결'
      : '사용 중';

  const diagnoses = getSheetObjects_(SHEETS.DIAG)
    .filter(row => String(row['케이스번호'] || '') === String(caseId))
    .sort((a, b) => tasteMindComparable_(b['검사일']) - tasteMindComparable_(a['검사일']));

  const preps = getSheetObjects_(SHEETS.PREPS)
    .filter(row => String(row['케이스번호'] || '') === String(caseId))
    .sort((a, b) => tasteMindComparable_(b['생성일']) - tasteMindComparable_(a['생성일']));

  const latestGoal = getTasteMindLatestManagerGoal_(caseId);
  const explicitGoal = String(options.goalText || '').trim();
  const managerGoal = explicitGoal || String(latestGoal['목표문장'] || '').trim();
  const oldMission = existing ? String(existing['최근실천목표'] || '').trim() : '';
  const currentMission = managerGoal || oldMission;

  const topic =
    String((preps[0] || {})['상담영역'] || '').trim() ||
    String(caseRow['주상담주제'] || '').trim();

  const area = inferTasteMindArea_(topic, diagnoses[0] || {});
  const meals = getExternalSheetObjects_(ss, TASTE_MIND_DEFAULTS.MEAL_SHEET);
  const completed = getTasteMindCompletedSessionCount_(caseId, meals);
  const journey = completed >= 5 ? 5 : completed + 1;

  const usedCodes = new Set(
    accounts.map(row => String(row['탐험코드'] || '').trim()).filter(Boolean)
  );

  const explorerCode =
    existing && existing['탐험코드']
      ? String(existing['탐험코드'])
      : generateTasteMindExplorerCode_(usedCodes);

  const accountId =
    existing && existing['계정ID']
      ? existing['계정ID']
      : 'EX-' + Utilities.getUuid().replace(/-/g, '').slice(0, 12).toUpperCase();

  const alias =
    existing && existing['탐험가별명']
      ? existing['탐험가별명']
      : makeTasteMindAlias_(student['학생코드'] + '|' + caseId);

  const token =
    existing && existing['접속토큰']
      ? existing['접속토큰']
      : Utilities.getUuid().replace(/-/g, '');

  const now = new Date();

  upsertTasteMindObject_(
    accountSheet,
    '케이스번호',
    caseId,
    {
      '계정ID': accountId,
      '학생코드': student['학생코드'] || '',
      '케이스번호': caseId,
      '학년도': student['학년도'] || '',
      '학년': student['학년'] || '',
      '반': student['반'] || '',
      '번호': student['번호'] || '',
      '학생명': student['학생명'] || '',
      '탐험가별명': alias,
      '탐험코드': explorerCode,
      '접속토큰': token,
      '내부상담영역': area,
      '학생표시활동': TASTE_MIND_ACTIVITY_LABELS[area] || TASTE_MIND_ACTIVITY_LABELS.common,
      '현재탐험': journey,
      '매니저현재단계': status,
      '최근실천목표': currentMission,
      '최근동기화': now,
      '계정상태': accountStatus,
      '코드발급일': existing ? existing['코드발급일'] : now,
      '코드재발급일': existing ? existing['코드재발급일'] : '',
      '최근접속': existing ? existing['최근접속'] : '',
      '비고': existing ? existing['비고'] : '매니저 자동 생성'
    }
  );

  if (managerGoal && accountStatus === '사용 중') {
    syncTasteMindGoal_(
      missionSheet,
      {
        accountId: accountId,
        studentCode: student['학생코드'] || '',
        caseId: caseId,
        journey: journey,
        text: managerGoal,
        targetCount: latestGoal['목표횟수'] || '',
        startDate: latestGoal['시작일'] || now,
        endDate: latestGoal['종료일'] || '',
        checkMethod: latestGoal['확인방법'] || '다음 상담에서 확인'
      }
    );
  }

  if (accountStatus === '종결') {
    closeTasteMindMissions_(missionSheet, caseId);
  }

  return {
    ok: true,
    created: !existing,
    updated: Boolean(existing),
    eligible: eligibility.eligible,
    accountId: accountId,
    explorerCode: explorerCode,
    accountStatus: accountStatus,
    completedSessionCount: completed,
    currentJourney: journey,
    currentMission: currentMission,
    webAppUrl: config.webAppUrl
  };
}

function getTasteMindEligibility_(caseId, caseRow, consent) {
  const c = caseRow || getSheetObjects_(SHEETS.CASES).find(row =>
    String(row['케이스번호'] || '') === String(caseId)
  ) || {};
  const g = consent || getSheetObjects_(SHEETS.CONSENTS).find(row =>
    String(row['케이스번호'] || '') === String(caseId)
  ) || {};

  if (String(c['현재단계'] || '') === '종결') {
    return { eligible: false, message: '상담이 종결되어 탐험소 계정을 생성하지 않습니다.' };
  }
  if (String(g['학생참여의사'] || '') === '참여하지 않음') {
    return { eligible: false, message: '학생 참여 의사가 “참여하지 않음”입니다.' };
  }
  if (String(g['확인상태'] || '') !== '동의 완료') {
    return { eligible: false, message: '보호자 동의 완료 후 탐험소 계정이 자동 생성됩니다.' };
  }
  if (String(g['학생참여의사'] || '') !== '참여 희망') {
    return { eligible: false, message: '학생 참여 의사를 “참여 희망”으로 저장하면 탐험소 계정이 자동 생성됩니다.' };
  }
  return { eligible: true, message: '탐험소 자동 등록 대상입니다.' };
}

function getTasteMindLatestManagerGoal_(caseId) {
  return getSheetObjects_(SHEETS.GOALS)
    .filter(row => String(row['케이스번호'] || '') === String(caseId))
    .sort((a, b) => {
      const dateDiff =
        tasteMindComparable_(b['시작일']) -
        tasteMindComparable_(a['시작일']);
      return dateDiff !== 0
        ? dateDiff
        : String(b['목표ID'] || '').localeCompare(String(a['목표ID'] || ''));
    })[0] || {};
}

function inferTasteMindArea_(topic, diagnosis) {
  const text = [
    topic,
    diagnosis['종합등급'],
    diagnosis['섭식태도'],
    diagnosis['체형인식'],
    diagnosis['알레르기'],
    diagnosis['추가상담요청사항']
  ].map(value => String(value || '').trim()).join(' ');

  if (/알레르기|알러지/.test(text)) return 'allergy';
  if (/편식|감각|새음식|음식거부|선택적/.test(text)) return 'picky';
  if (/저체중|섭취부족|소식|체중증가|먹는양부족/.test(text)) return 'strength';
  if (/성장|키|성장부진/.test(text)) return 'growth';
  if (/비만|과체중|체중조절|생활리듬|과식|빠른식사|결식|아침식사/.test(text)) return 'rhythm';
  return 'common';
}

function getTasteMindCompletedSessionCount_(caseId, rows) {
  const dates = new Set();
  (rows || [])
    .filter(row => String(row['케이스번호'] || '') === String(caseId))
    .forEach(row => {
      const key = tasteMindDateKey_(row['급식일'] || row['기록일시']);
      if (key) dates.add(key);
    });
  return Math.min(5, dates.size);
}

function tasteMindDateKey_(value) {
  if (!value) return '';
  const date = Object.prototype.toString.call(value) === '[object Date]'
    ? value
    : new Date(value);
  if (!isNaN(date.getTime())) {
    return Utilities.formatDate(date, TASTE_MIND_DEFAULTS.TIMEZONE, 'yyyy-MM-dd');
  }
  const digits = String(value).replace(/[^0-9]/g, '');
  return digits.length >= 8
    ? digits.slice(0, 4) + '-' + digits.slice(4, 6) + '-' + digits.slice(6, 8)
    : '';
}

function generateTasteMindExplorerCode_(usedCodes) {
  for (let i = 0; i < 10000; i += 1) {
    const code = String(Math.floor(1000 + Math.random() * 9000));
    if (!usedCodes.has(code)) return code;
  }
  throw new Error('사용 가능한 4자리 탐험코드를 생성하지 못했습니다.');
}

function makeTasteMindAlias_(seed) {
  const first = ['민트','햇살','초록','구름','별빛','바람','푸른','도토리','새싹','달빛'];
  const second = ['별','콩','새','곰','토끼','나무','여우','고래','씨앗','나비'];
  let hash = 0;
  const text = String(seed || '');
  for (let i = 0; i < text.length; i += 1) {
    hash = ((hash << 5) - hash) + text.charCodeAt(i);
    hash |= 0;
  }
  return first[Math.abs(hash) % first.length] +
    second[Math.abs(Math.floor(hash / first.length)) % second.length];
}

    '어디서': '학교 또는 가정',

    '도움받을사람': '영양선생님·담임선생님·보호자',
    '확인방법': payload.checkMethod,

    '종료일': payload.endDate || '',
    '상태': '진행 중',
    '종료일시': '',
    '비고': 'AI 영양상담 매니저에서 자동 전송'

function markTasteMindLinkReflected_(caseId, linkId, sessionId) {
  if (!linkId) return { ok: true, reflected: false };

  const ss = SpreadsheetApp.openById(getTasteMindConfig_().spreadsheetId);
  const sheet = ss.getSheetByName(TASTE_MIND_DEFAULTS.LINK_SHEET);
  if (!sheet) throw new Error('탐험소 매니저연계 시트가 없습니다.');

  const row = getTasteMindObjects_(sheet).find(item =>
    String(item['연계ID'] || '') === String(linkId) &&
    String(item['케이스번호'] || '') === String(caseId)
  );
  if (!row) throw new Error('반영할 탐험소 활동을 찾지 못했습니다.');

  updateTasteMindObject_(sheet, '연계ID', linkId, {
    '반영상태': '반영완료',
    '반영일시': new Date(),
    '비고': appendTasteMindText_(row['비고'], '매니저 상담회기 ' + sessionId + '에 반영')
  });

  return { ok: true, reflected: true, linkId: linkId };
}

function getTasteMindObjects_(sheet) {
  const values = sheet.getDataRange().getValues();
  if (values.length < 2) return [];
  const headers = values[0].map(value => String(value || '').trim());
  return values.slice(1)
    .filter(row => row.some(value => value !== '' && value !== null))
    .map(row => {
      const object = {};
      headers.forEach((header, index) => {
        if (header) object[header] = row[index];
      });
      return object;
    });
}

function upsertTasteMindObject_(sheet, keyHeader, keyValue, patch) {
  const data = sheet.getDataRange().getValues();
  const headers = data[0].map(value => String(value || '').trim());
  const keyCol = headers.indexOf(keyHeader);
  if (keyCol < 0) throw new Error(sheet.getName() + ' 시트에 ' + keyHeader + ' 열이 없습니다.');

  let target = -1;
  for (let i = 1; i < data.length; i += 1) {
    if (String(data[i][keyCol] || '') === String(keyValue)) {
      target = i + 1;
      break;
    }
  }

  const oldRow = target > 0 ? data[target - 1] : [];
  const row = headers.map((header, index) =>
    Object.prototype.hasOwnProperty.call(patch, header)
      ? patch[header]
      : (target > 0 ? oldRow[index] : '')
  );

  if (target > 0) sheet.getRange(target, 1, 1, headers.length).setValues([row]);
  else sheet.appendRow(row);
}

function updateTasteMindObject_(sheet, keyHeader, keyValue, patch) {
  const data = sheet.getDataRange().getValues();
  const headers = data[0].map(value => String(value || '').trim());
  const keyCol = headers.indexOf(keyHeader);
  if (keyCol < 0) throw new Error(sheet.getName() + ' 시트에 ' + keyHeader + ' 열이 없습니다.');

  for (let i = 1; i < data.length; i += 1) {
    if (String(data[i][keyCol] || '') !== String(keyValue)) continue;
    const row = headers.map((header, index) =>
      Object.prototype.hasOwnProperty.call(patch, header)
        ? patch[header]
        : data[i][index]
    );
    sheet.getRange(i + 1, 1, 1, headers.length).setValues([row]);
    return true;
  }
  return false;
}

function appendTasteMindObject_(sheet, object) {
  const headers = sheet
    .getRange(1, 1, 1, sheet.getLastColumn())
    .getValues()[0]
    .map(value => String(value || '').trim());

  sheet.appendRow(headers.map(header =>
    Object.prototype.hasOwnProperty.call(object, header)
      ? object[header]
      : ''
  ));
}

function appendTasteMindText_(original, addition) {
  return [String(original || '').trim(), String(addition || '').trim()]
    .filter(Boolean)
    .join(' | ');
}

function getDashboardData() {
  ensureDashboardScheduleStructure_();
  ensureDashboardCompletionStructure_();

  const cases = getSheetObjects_(SHEETS.CASES);
  const statuses = CASE_STATUS_VALUES.slice();
  const counts = {};

  statuses.forEach(status => {
    counts[status] = 0;
  });

  cases.forEach(row => {
    const status = row['현재단계'];

    if (status in counts) {
      counts[status] += 1;
    }
  });

  counts['신규 접수'] =
    listPendingIntakes().length;

  const now = new Date();
  const todayKey =
    dashboardDateKey_(now);

  const upcomingEnd = new Date(
    now.getFullYear(),
    now.getMonth() + 4,
    0
  );

  const upcomingEvents =
    listDashboardEvents_(
      todayKey,
      dashboardDateKey_(upcomingEnd)
    )
      .filter(event =>
        event.date >= todayKey
      )
      .sort(compareDashboardEvents_)
      .slice(0, 12);

  const todayEvents =
    listDashboardEvents_(
      todayKey,
      todayKey
    );

  return {
    counts: counts,
    total: cases.length,
    dueToday: todayEvents.length,
    upcoming: upcomingEvents,
    calendar: getDashboardCalendar(
      now.getFullYear(),
      now.getMonth() + 1
    )
  };
}

function ensureDashboardScheduleStructure_() {
  const sheet = ensureSheetHeaders_(
    SHEETS.SCHEDULES,
    DASHBOARD_SCHEDULE_HEADERS
  );

  const headers = sheet
    .getRange(
      1,
      1,
      1,
      sheet.getLastColumn()
    )
    .getValues()[0];

  const typeCol =
    headers.indexOf('일정유형');

  if (typeCol >= 0) {
    const rule = SpreadsheetApp
      .newDataValidation()
      .requireValueInList(
        DASHBOARD_SCHEDULE_TYPES,
        true
      )
      .setAllowInvalid(false)
      .build();

    sheet.getRange(
      2,
      typeCol + 1,
      Math.max(
        sheet.getMaxRows() - 1,
        1
      ),
      1
    ).setDataValidation(rule);
  }

  const dateCol =
    headers.indexOf('일정일');

  if (dateCol >= 0) {
    sheet.getRange(
      2,
      dateCol + 1,
      Math.max(
        sheet.getMaxRows() - 1,
        1
      ),
      1
    ).setNumberFormat('yyyy-mm-dd');
  }

  const timeHeaders = [
    '시작시간',
    '종료시간'
  ];

  timeHeaders.forEach(header => {
    const column = headers.indexOf(header);

    if (column >= 0) {
      sheet.getRange(
        2,
        column + 1,
        Math.max(
          sheet.getMaxRows() - 1,
          1
        ),
        1
      ).setNumberFormat('@');
    }
  });

  return sheet;
}

function ensureDashboardCompletionStructure_() {
  const sheet = ensureSheetHeaders_(
    SHEETS.SCHEDULE_DONE,
    DASHBOARD_COMPLETION_HEADERS
  );

  const headers = sheet
    .getRange(
      1,
      1,
      1,
      sheet.getLastColumn()
    )
    .getValues()[0];

  const dateCol =
    headers.indexOf('일정일');

  if (dateCol >= 0) {
    sheet.getRange(
      2,
      dateCol + 1,
      Math.max(
        sheet.getMaxRows() - 1,
        1
      ),
      1
    ).setNumberFormat('yyyy-mm-dd');
  }

  const doneDateCol =
    headers.indexOf('완료일시');

  if (doneDateCol >= 0) {
    sheet.getRange(
      2,
      doneDateCol + 1,
      Math.max(
        sheet.getMaxRows() - 1,
        1
      ),
      1
    ).setNumberFormat('yyyy-mm-dd hh:mm');
  }

    console.log(
      '일정완료 시트 숨김 생략: ' +
      error.message
    );
  }

  return sheet;
}

function getDashboardCalendar(year, month) {
  ensureDashboardScheduleStructure_();
  ensureDashboardCompletionStructure_();

  const normalizedYear =
    Number(year) ||
    new Date().getFullYear();

  const normalizedMonth =
    Math.max(
      1,
      Math.min(
        12,
        Number(month) ||
        new Date().getMonth() + 1
      )
    );

  const startDate = new Date(
    normalizedYear,
    normalizedMonth - 1,
    1
  );

  const endDate = new Date(
    normalizedYear,
    normalizedMonth,
    0
  );

  const startKey =
    dashboardDateKey_(startDate);

  const endKey =
    dashboardDateKey_(endDate);

  const events =
    listDashboardEvents_(
      startKey,
      endKey
    ).sort(compareDashboardEvents_);

  const days = {};

  for (
    let day = 1;
    day <= endDate.getDate();
    day += 1
  ) {
    const dateKey =
      dashboardDateKey_(
        new Date(
          normalizedYear,
          normalizedMonth - 1,
          day
        )
      );

    days[dateKey] = {
      date: dateKey,
      events: [],
      appointmentCount: 0,
      hasTrip: false,
      isFull: false
    };
  }

  events.forEach(event => {
    if (!days[event.date]) return;

    days[event.date].events.push(event);

    if (
      DASHBOARD_APPOINTMENT_TYPES
        .includes(event.type)
    ) {
      days[event.date]
        .appointmentCount += 1;
    }

    if (event.type === '출장') {
      days[event.date].hasTrip = true;
    }
  });

  Object.keys(days).forEach(dateKey => {
    days[dateKey].isFull =
      days[dateKey]
        .appointmentCount >=
      DASHBOARD_DAILY_APPOINTMENT_LIMIT;
  });

  return toClient_({
    year: normalizedYear,
    month: normalizedMonth,
    monthLabel:
      normalizedYear +
      '년 ' +
      normalizedMonth +
      '월',
    firstWeekday:
      startDate.getDay(),
    lastDay:
      endDate.getDate(),
    today:
      dashboardDateKey_(
        new Date()
      ),
    dailyLimit:
      DASHBOARD_DAILY_APPOINTMENT_LIMIT,
    days: days
  });
}

function saveDashboardManualSchedule(payload) {
  payload = payload || {};

  const scheduleDate =
    dashboardDateKey_(
      payload.date
    );

  const scheduleType =
    String(
      payload.type || ''
    ).trim();

  let title =
    String(
      payload.title || ''
    ).trim();

  if (!scheduleDate) {
    throw new Error(
      '일정 날짜를 입력하세요.'
    );
  }

  if (
    !DASHBOARD_SCHEDULE_TYPES
      .includes(scheduleType)
  ) {
    throw new Error(
      '일정 유형을 선택하세요.'
    );
  }

  if (!title) {
    title =
      scheduleType === '출장'
        ? '출장'
        : scheduleType;
  }

  const startTime =
    normalizeDashboardTime_(
      payload.startTime
    );

  const endTime =
    normalizeDashboardTime_(
      payload.endTime
    );

  if (
    startTime &&
    endTime &&
    startTime >= endTime
  ) {
    throw new Error(
      '종료시간은 시작시간보다 늦어야 합니다.'
    );
  }

  if (
    DASHBOARD_APPOINTMENT_TYPES
      .includes(scheduleType)
  ) {
    assertDashboardAppointmentCapacity_(
      scheduleDate,
      '',
      ''
    );
  }

  const scheduleId =
    nextId_(
      SHEETS.SCHEDULES,
      'CAL',
      '일정ID',
      4
    );

  const now = new Date();

  appendObject_(
    SHEETS.SCHEDULES,
    {
      '일정ID': scheduleId,
      '일정일':
        safeDateForSheet_(
          scheduleDate,
          ''
        ),
      '시작시간': startTime,
      '종료시간': endTime,
      '일정유형': scheduleType,
      '제목': title,
      '케이스번호':
        String(
          payload.caseId || ''
        ).trim(),
      '학생명':
        String(
          payload.studentName || ''
        ).trim(),
      '메모':
        String(
          payload.note || ''
        ).trim(),
      '생성일시': now,

function listDashboardEvents_(
  startKey,
  endKey

  const completionMap =
    getDashboardCompletionMap_();

  getSheetObjects_(SHEETS.CASES)
    .forEach(caseRow => {

        dashboardDateKey_(
          caseRow['다음일정']
        );

        !dateKey ||
        dateKey < startKey ||
        dateKey > endKey
      ) {
        return;
      }

        );

      const student =
        studentMap[

        student['학생명'] || '';

        buildDashboardCompletionKey_(
          'case',
          caseId,
          dateKey
        );

        completionMap[completionKey] || {};

      events.push({
        id:
          'CASE-' +

        source: 'case',

        completionKey: completionKey,
        caseId: caseId,

        endTime: '',

          studentName

            : '상담 일정',

        gradeClass:
          formatGradeClass_(student),
        topic:
          caseRow['주상담주제'] || '',
        status:
          caseRow['현재단계'] || '',
        note:
          '상담케이스의 다음일정에서 자동 표시',
        editable: false,
        completed:
          completion.completed === true,
        completedAt:
          completion.completedAt || ''
      });
    });

  ensureDashboardScheduleStructure_();

  getSheetObjects_(SHEETS.SCHEDULES)
    .forEach(row => {
      const dateKey =
        dashboardDateKey_(
          row['일정일']
        );

      if (
        !dateKey ||
        dateKey < startKey ||
        dateKey > endKey
      ) {
        return;
      }

        String(
          row['일정ID'] || ''
        );

        buildDashboardCompletionKey_(
          'manual',

          dateKey
        );

        completionMap[completionKey] || {};

        id: scheduleId,
        source: 'manual',
        sourceId: scheduleId,

        caseId:

        status: '',

          row['메모'] || '',
        editable: true,
        completed:
          completion.completed === true,
        completedAt:
          completion.completedAt || ''
      });
    });

  return events;
}

function buildDashboardCompletionKey_(
  source,
  sourceId,
  dateKey
) {
  const normalizedSource =
    String(source || '').trim();

  const normalizedId =
    String(sourceId || '').trim();

  const normalizedDate =
    dashboardDateKey_(dateKey);

  if (
    !normalizedSource ||
    !normalizedId ||
    !normalizedDate
  ) {
    return '';
  }

  return [
    normalizedSource,
    normalizedId,
    normalizedDate
  ].join('|');
}

function getDashboardCompletionMap_() {
  ensureDashboardCompletionStructure_();

  const map = {};

  getSheetObjects_(
    SHEETS.SCHEDULE_DONE
  ).forEach(row => {
    const key =
      String(
        row['완료키'] || ''
      ).trim();

    if (!key) return;

    map[key] = {
      completed:
        String(
          row['완료여부'] || ''
        ) === '완료',
      completedAt:
        formatDateTime_(
          row['완료일시']
        )
    };
  });

  return map;
}

/**
 * 달력의 체크박스를 선택하거나 해제하면 완료상태를 저장합니다.
 * 상담케이스 자동 일정과 직접 입력 일정 모두 지원합니다.
 */
function toggleDashboardScheduleCompletion(
  payload
) {
  payload = payload || {};

  const source =
    String(
      payload.source || ''
    ).trim();

  const sourceId =
    String(
      payload.sourceId || ''
    ).trim();

  const dateKey =
    dashboardDateKey_(
      payload.date
    );

  const title =
    String(
      payload.title || ''
    ).trim();

  const caseId =
    String(
      payload.caseId || ''
    ).trim();

  const completed =
    payload.completed === true ||
    String(payload.completed) === 'true';

  if (
    !['case', 'manual']
      .includes(source)
  ) {
    throw new Error(
      '일정 원본정보가 올바르지 않습니다.'
    );
  }

  if (!sourceId || !dateKey) {
    throw new Error(
      '일정 식별정보가 부족합니다.'
    );
  }

  /*
   * 오래 열린 화면에서 이미 삭제되거나 변경된 일정을
   * 체크하지 않도록 원본 존재 여부를 확인합니다.
   */
  if (source === 'case') {
    const caseRow =
      getSheetObjects_(
        SHEETS.CASES
      ).find(row =>
        String(
          row['케이스번호'] || ''
        ) === sourceId
      );

    if (!caseRow) {
      throw new Error(
        '해당 상담케이스를 찾지 못했습니다. 화면을 새로고침해 주세요.'
      );
    }

    const currentDate =
      dashboardDateKey_(
        caseRow['다음일정']
      );

    if (currentDate !== dateKey) {
      throw new Error(
        '상담 일정 날짜가 변경되었습니다. 화면을 새로고침해 주세요.'
      );
    }
  }

  if (source === 'manual') {
    const scheduleRow =
      getSheetObjects_(
        SHEETS.SCHEDULES
      ).find(row =>
        String(
          row['일정ID'] || ''
        ) === sourceId
      );

    if (!scheduleRow) {
      throw new Error(
        '직접 입력한 일정을 찾지 못했습니다. 화면을 새로고침해 주세요.'
      );
    }
  }

  ensureDashboardCompletionStructure_();

  const completionKey =
    buildDashboardCompletionKey_(
      source,
      sourceId,
      dateKey
    );

  const now = new Date();

  updateOrAppendByKey_(
    SHEETS.SCHEDULE_DONE,
    '완료키',
    completionKey,
    {
      '완료키': completionKey,
      '원본구분': source,
      '원본ID': sourceId,
      '케이스번호': caseId,
      '일정일':
        safeDateForSheet_(
          dateKey,
          ''
        ),
      '일정명': title,
      '완료여부':
        completed
          ? '완료'
          : '미완료',
      '완료일시':
        completed
          ? now
          : '',
      '수정일시': now
    }
  );

  log_(
    'UPDATE',
    SHEETS.SCHEDULE_DONE,
    completionKey,
    [
      title || sourceId,
      completed
        ? '완료 체크'
        : '완료 체크 해제'
    ].join(' | ')
  );

  const date =
    asDate_(dateKey) ||
    new Date();

  return {
    ok: true,
    completed: completed,
    calendar: getDashboardCalendar(
      date.getFullYear(),
      date.getMonth() + 1
    ),
    dashboard: getDashboardData()
  };
}

function compareDashboardEvents_(a, b) {
  const left = [
    a.date || '',
    a.completed ? '1' : '0',
    a.startTime || '99:99',
    a.type || '',
    a.title || ''
  ].join('|');

  const right = [
    b.date || '',
    b.completed ? '1' : '0',
    b.startTime || '99:99',
    b.type || '',
    b.title || ''
  ].join('|');

  return left.localeCompare(right);
}

function assertDashboardAppointmentCapacity_(
  dateValue,
  excludeCaseId,
  excludeScheduleId
) {
  const dateKey =
    dashboardDateKey_(
      dateValue
    );

  if (!dateKey) return;

  let count = 0;

  getSheetObjects_(SHEETS.CASES)
    .forEach(row => {
      if (
        excludeCaseId &&
        String(row['케이스번호'] || '') ===
        String(excludeCaseId)
      ) {
        return;
      }

      if (
        dashboardDateKey_(
          row['다음일정']
        ) === dateKey
      ) {
        count += 1;
      }
    });

  ensureDashboardScheduleStructure_();

  getSheetObjects_(SHEETS.SCHEDULES)
    .forEach(row => {
      if (
        excludeScheduleId &&
        String(row['일정ID'] || '') ===
        String(excludeScheduleId)
      ) {
        return;
      }

      if (
        dashboardDateKey_(
          row['일정일']
        ) === dateKey &&
        DASHBOARD_APPOINTMENT_TYPES
          .includes(
            String(
              row['일정유형'] || ''
            )
          )
      ) {
        count += 1;
      }
    });

  if (
    count >=
    DASHBOARD_DAILY_APPOINTMENT_LIMIT
  ) {
    throw new Error(
      dateKey +
      '에는 상담·사전방문 일정이 이미 ' +
      count +
      '건 있습니다. 하루 최대 ' +
      DASHBOARD_DAILY_APPOINTMENT_LIMIT +
      '건까지만 등록할 수 있습니다.'
    );
  }
}

function dashboardDateKey_(value) {
  if (!value) return '';

  const date = asDate_(value);

  if (!date) return '';

  return Utilities.formatDate(
    date,
    Session.getScriptTimeZone(),
    'yyyy-MM-dd'
  );
}

function normalizeDashboardTime_(value) {
  if (!value) return '';

  if (
    Object.prototype.toString.call(value) ===
    '[object Date]'
  ) {
    return Utilities.formatDate(
      value,
      Session.getScriptTimeZone(),
      'HH:mm'
    );
  }

  const text =
    String(value).trim();

  const match =
    text.match(
      /^(\d{1,2}):(\d{2})/
    );

  if (!match) return '';

  const hour =
    Number(match[1]);

  const minute =
    Number(match[2]);

  if (
    hour < 0 ||
    hour > 23 ||
    minute < 0 ||
    minute > 59
  ) {
    return '';
  }

  return (
    Utilities.formatString(
      '%02d',
      hour
    ) +
    ':' +
    Utilities.formatString(
      '%02d',
      minute
    )
  );
}

function approveIntake(intakeId) {
  const lock = LockService.getScriptLock();
  lock.waitLock(30000);
  try {
    const row = getSheetObjects_(SHEETS.INTAKE).find(r => String(r['접수ID']) === String(intakeId));
    if (!row) throw new Error('접수 건을 찾을 수 없습니다.');
    if (String(row['처리상태']) === '케이스 생성') {
      return { ok: true, message: '이미 상담 케이스로 전환된 접수입니다.' };
    }

    const result = createCaseFromIntakeRow_(row);
    updateRowByKey_(SHEETS.INTAKE, '접수ID', intakeId, {
      '처리상태': '케이스 생성',
      '학생코드': result.studentCode,
      '비고': appendText_(row['비고'], '케이스번호: ' + result.caseId)
    });
    log_('APPROVE', SHEETS.INTAKE, intakeId, '공개 접수 승인 및 케이스 생성');
    return result;
  } finally {
    lock.releaseLock();
  }
}

function listCases(filters) {
  filters = filters || {};
  const studentMap = studentMap_();
  const sessionMap = {};
  const goalMap = {};

  getSheetObjects_(SHEETS.SESSIONS).forEach(row => {
    const key = String(row['케이스번호'] || '');
    if (!sessionMap[key]) sessionMap[key] = [];
    sessionMap[key].push(row);
  });

  getSheetObjects_(SHEETS.GOALS).forEach(row => {
    const key = String(row['케이스번호'] || '');
    if (!goalMap[key]) goalMap[key] = [];
    goalMap[key].push(row);
  });

  let rows = getSheetObjects_(SHEETS.CASES).map(r => {
    const caseId = String(r['케이스번호'] || '');
    const s = studentMap[r['학생코드']] || {};
    const sessions = (sessionMap[caseId] || []).slice().sort((a, b) => {
      const noDiff = Number(a['회기'] || 0) - Number(b['회기'] || 0);
      if (noDiff !== 0) return noDiff;
      const ad = asDate_(a['상담일']);
      const bd = asDate_(b['상담일']);
      return (ad ? ad.getTime() : 0) - (bd ? bd.getTime() : 0);
    });
    const goals = goalMap[caseId] || [];
    const latestSession = sessions.length ? sessions[sessions.length - 1] : {};
    const latestGoal = goals.length ? goals[goals.length - 1] : {};

    return {
      caseId,
      studentCode: r['학생코드'],
      studentName: s['학생명'] || '',
      gradeClass: formatGradeClass_(s),
      intakeDate: formatDate_(r['접수일']),
      referral: r['신청경로'],
      topic: r['주상담주제'],
      status: r['현재단계'],
      sessionCount: sessions.length,
      lastSessionDate: formatDate_(latestSession['상담일']),
      nextDate: formatDate_(r['다음일정']),
      latestGoal: latestGoal['목표문장'] || '',
      goalResult: latestGoal['결과'] || '',
      teacher: r['담당자'],
      folderUrl: r['Drive폴더URL']
    };
  });

  if (filters.keyword) {
    const q = String(filters.keyword).trim().toLowerCase();
    rows = rows.filter(r => Object.values(r).some(v => String(v || '').toLowerCase().includes(q)));
  }
  if (filters.status) rows = rows.filter(r => r.status === filters.status);
  if (filters.topic) rows = rows.filter(r => r.topic === filters.topic);
  return rows.reverse();
}

function getCaseDetailRaw_(caseId) {
  const findBy = (sheetName, key) => getSheetObjects_(sheetName).filter(r => String(r['케이스번호']) === String(key));
  const caseRow = findBy(SHEETS.CASES, caseId)[0];
  if (!caseRow) throw new Error('상담 케이스를 찾을 수 없습니다.');
  const student = getSheetObjects_(SHEETS.STUDENTS).find(r => r['학생코드'] === caseRow['학생코드']) || {};
  return {
    case: caseRow,
    student,
    consents: findBy(SHEETS.CONSENTS, caseId),
    diagnostics: findBy(SHEETS.DIAG, caseId),
    sessions: findBy(SHEETS.SESSIONS, caseId),
    goals: findBy(SHEETS.GOALS, caseId),
    goalChecks: findBy(SHEETS.GOAL_CHECKS, caseId),
    preparations: findBy(SHEETS.PREPS, caseId),
    documents: findBy(SHEETS.DOCS, caseId)
  };
}

    const ad = asDate_(a['상담일']);
    const bd = asDate_(b['상담일']);

function generateConsentLink(caseId) {
  validateRequired_({ caseId }, ['caseId']);

  const settings = getSettings_();
  const baseUrl = String(
    settings.PUBLIC_APP_URL || ''
  ).trim().split('?')[0];

  if (!baseUrl) {
    throw new Error(
      '설정 시트의 PUBLIC_APP_URL에 공개 접수·동의 포털 웹앱 주소를 입력하세요.'
    );
  }

  /*
   * 기존 보호자동의 시트에 '짧은코드' 열이 없어도 자동으로 추가합니다.
   */
  ensureSheetHeaders_(
    SHEETS.CONSENTS,
    ['짧은코드']
  );

  const current = getSheetObjects_(
    SHEETS.CONSENTS
  ).find(row =>
    String(row['케이스번호']) ===
    String(caseId)
  ) || {};

  /*
   * 긴 토큰은 내부 식별값으로 계속 보관합니다.
   * 문자로 보내는 주소에는 짧은코드만 노출합니다.
   */
  const token =
    current['동의토큰'] ||
    Utilities.getUuid().replace(/-/g, '');

  const shortCode =
    current['짧은코드'] ||
    generateUniqueConsentShortCode_();

  const shortLink =
    baseUrl +
    '?k=' +
    encodeURIComponent(shortCode);

  const legacyLink =
    baseUrl +
    '?mode=consent&token=' +
    encodeURIComponent(token);

  updateOrAppendByKey_(
    SHEETS.CONSENTS,
    '케이스번호',
    caseId,
    {
      '케이스번호': caseId,
      '동의토큰': token,
      '짧은코드': shortCode,
      '동의링크': shortLink,
      '확인상태':
        current['확인상태'] &&
        current['확인상태'] !== '미발송'
          ? current['확인상태']
          : '미발송',
      '보호자연락처':
        normalizeContactValue_(
          current['보호자연락처']
        ),
      '학생참여의사':
        current['학생참여의사'] ||
        '미확인'
    }
  );

  log_(
    'CREATE',
    SHEETS.CONSENTS,
    caseId,
    '보호자 동의 문자용 짧은 링크 생성'
  );

  return {
    ok: true,
    link: shortLink,
    shortLink: shortLink,
    legacyLink: legacyLink,
    shortCode: shortCode,
    token: token
  };
}

function generateUniqueConsentShortCode_() {
  const alphabet =
    'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';

  const existing = new Set(
    getSheetObjects_(SHEETS.CONSENTS)
      .map(row =>
        String(row['짧은코드'] || '')
          .trim()
          .toUpperCase()
      )
      .filter(Boolean)
  );

  for (let attempt = 0; attempt < 1000; attempt += 1) {
    let code = '';

    for (let index = 0; index < 10; index += 1) {
      code += alphabet.charAt(
        Math.floor(Math.random() * alphabet.length)
      );
    }

    if (!existing.has(code)) return code;
  }

  throw new Error(
    '짧은 동의코드를 생성하지 못했습니다. 잠시 후 다시 시도하세요.'
  );
}

  validateRequired_({ caseId }, ['caseId']);

  ensureSheetHeaders_(

}

function markConsentSent(caseId) {
  validateRequired_({ caseId }, ['caseId']);

  const consent = getSheetObjects_(SHEETS.CONSENTS)
    .find(r => String(r['케이스번호']) === String(caseId)) || {};

  if (!consent['동의링크']) {
    throw new Error('먼저 동의 링크를 생성하세요.');
  }

  const currentStatus = String(consent['확인상태'] || '미발송');
  const requiredConsentFields = [
    '상담동의',
    '개인정보동의',
    '민감정보동의',
    '진단결과활용동의'
  ];

  const allAgreed = requiredConsentFields.every(
    key => String(consent[key] || '') === '동의'
  );

  const hasGuardianSubmission = Boolean(
    consent['제출일시'] ||
    consent['동의일'] ||
    consent['동의서파일URL'] ||
    consent['보호자명']
  );

  if (hasGuardianSubmission && allAgreed) {
    if (currentStatus !== '동의 완료') {
      updateRowByKey_(SHEETS.CONSENTS, '케이스번호', caseId, {
        '확인상태': '교사 확인 필요'
      });
    }

    return {
      ok: true,
      status: currentStatus === '동의 완료' ? '동의 완료' : '교사 확인 필요',
      message: currentStatus === '동의 완료'
        ? '이미 보호자 동의 확인이 완료되었습니다.'
        : '보호자가 이미 동의서를 제출했습니다. 동의 내용을 확인한 뒤 동의 완료 처리해 주세요.'
    };
  }

  if (currentStatus === '동의 완료') {
    return {
      ok: true,
      status: '동의 완료',
      message: '이미 보호자 동의 확인이 완료되었습니다.'
    };
  }

  if (currentStatus === '비동의') {
    throw new Error('보호자가 상담에 동의하지 않은 건입니다.');
  }

  updateRowByKey_(SHEETS.CONSENTS, '케이스번호', caseId, {
    '확인상태': '동의 요청',
    '발송일': consent['발송일'] || new Date()
  });

  log_('UPDATE', SHEETS.CONSENTS, caseId, '보호자 동의 요청 발송 처리');

  return {
    ok: true,
    status: '동의 요청',
    message: '보호자에게 동의 링크를 전달한 것으로 처리했습니다.'
  };
}

function confirmGuardianConsent(caseId) {
  validateRequired_({ caseId }, ['caseId']);

  const consent = getSheetObjects_(SHEETS.CONSENTS)
    .find(r => String(r['케이스번호']) === String(caseId));

  if (!consent) {
    throw new Error('보호자 동의 정보를 찾을 수 없습니다.');
  }

  const currentStatus = String(consent['확인상태'] || '');

  if (currentStatus === '동의 완료') {
    const caseRow = getSheetObjects_(SHEETS.CASES)
      .find(r => String(r['케이스번호']) === String(caseId));

    if (caseRow && String(caseRow['현재단계'] || '') === '동의 대기') {
      updateCaseStatus(caseId, '진단 대기', '');
    }

    const explorerSync = syncTasteMindCaseSafe_(caseId, {
      allowCreate: true,
      source: '기존 동의 완료 확인'
    });

    return {
      ok: true,
      status: '동의 완료',
      message: '이미 동의 확인이 완료된 건입니다.',
      explorerSync: explorerSync
    };
  }

  if (currentStatus === '비동의') {
    throw new Error('보호자가 상담에 동의하지 않은 건입니다.');
  }

  const requiredConsentFields = [
    '상담동의',
    '개인정보동의',
    '민감정보동의',
    '진단결과활용동의'
  ];

  const missing = requiredConsentFields.filter(
    key => String(consent[key] || '') !== '동의'
  );

  if (missing.length) {
    throw new Error('필수 동의 항목을 확인하세요: ' + missing.join(', '));
  }

  const hasGuardianSubmission = Boolean(
    consent['제출일시'] ||
    consent['동의일'] ||
    consent['동의서파일URL'] ||
    consent['보호자명']
  );

  if (!hasGuardianSubmission) {
    throw new Error('보호자가 제출한 동의서가 확인되지 않습니다. 제출일이나 동의서 PDF를 확인하세요.');
  }

  updateRowByKey_(SHEETS.CONSENTS, '케이스번호', caseId, {
    '확인상태': '동의 완료',
    '확인일': new Date(),
    '확인자': Session.getActiveUser().getEmail()
  });

  updateCaseStatus(caseId, '진단 대기', '');

  const explorerSync = syncTasteMindCaseSafe_(caseId, {
    allowCreate: true,
    source: '보호자 동의 완료'
  });

  log_('VERIFY', SHEETS.CONSENTS, caseId, '보호자 동의 교사 확인 완료');

  return {
    ok: true,
    status: '동의 완료',
    message: '동의 확인을 완료했습니다. 학생 참여 희망이 확인되면 탐험소 계정도 자동 등록됩니다.',
    explorerSync: explorerSync
  };
}

function getDiagnosisRowsForCase_(caseId) {
  ensureV36Structure_();
  return getSheetObjects_(SHEETS.DIAG)
    .filter(r => String(r['케이스번호']) === String(caseId))
    .sort((a, b) => {
      const noDiff = Number(a['검사차수'] || 0) - Number(b['검사차수'] || 0);
      if (noDiff !== 0) return noDiff;
      const ad = asDate_(a['검사일']);
      const bd = asDate_(b['검사일']);
      return (ad ? ad.getTime() : 0) - (bd ? bd.getTime() : 0);
    });
}

function getPrimaryDiagnosis_(detail) {
  const rows = (detail.diagnostics || []).slice();
  return rows.find(r => String(r['평가시점'] || '') === '사전') || rows[0] || {};
}

function getLatestDiagnosis_(detail) {
  const rows = (detail.diagnostics || []).slice().sort((a, b) => {
    const noDiff = Number(a['검사차수'] || 0) - Number(b['검사차수'] || 0);
    if (noDiff !== 0) return noDiff;
    const ad = asDate_(a['검사일']);
    const bd = asDate_(b['검사일']);
    return (ad ? ad.getTime() : 0) - (bd ? bd.getTime() : 0);
  });
  return rows.length ? rows[rows.length - 1] : {};
}

function ensureDiagnosisRecord_(caseId, evaluationPoint) {
  ensureV36Structure_();
  const point = String(evaluationPoint || '사전').trim() || '사전';
  const rows = getDiagnosisRowsForCase_(caseId);
  const samePoint = rows.find(r => String(r['평가시점'] || '') === point);
  if (samePoint) return samePoint;

  const detail = getCaseDetailRaw_(caseId);
  const no = rows.reduce((m, r) => Math.max(m, Number(r['검사차수'] || 0)), 0) + 1;
  const diagnosisId = 'DG-' + caseId + '-' + Utilities.formatString('%02d', no);

  const row = {
    '진단ID': diagnosisId,
    '케이스번호': caseId,
    '검사차수': no,
    '평가시점': point,
    '학생코드': detail.case['학생코드'],
    '학생명': detail.student['학생명'] || '',
    '학년도': detail.student['학년도'] || '',
    '학년': detail.student['학년'] || '',
    '반': detail.student['반'] || '',
    '교사확인': '미확인'
  };

  appendObject_(SHEETS.DIAG, row);
  return row;
}

function getDiagnosisRecordById_(diagnosisId) {
  ensureV36Structure_();
  return getSheetObjects_(SHEETS.DIAG)
    .find(r => String(r['진단ID']) === String(diagnosisId)) || {};
}

function uploadCaseFile(payload) {
  validateRequired_(payload, ['caseId', 'fileType', 'fileName', 'mimeType', 'base64']);
  if (payload.fileType === '진단결과' || payload.fileType === '응답내역') {
    assertConsentComplete_(payload.caseId);
  }

  const detail = getCaseDetailRaw_(payload.caseId);
  const folderId = extractDriveId_(detail.case['Drive폴더URL']);
  if (!folderId) throw new Error('Drive 폴더 정보를 찾을 수 없습니다.');

  const caseFolder = DriveApp.getFolderById(folderId);
  const folderMap = {
    '동의서': '02_보호자동의',
    '진단결과': '03_공식진단',
    '응답내역': '03_공식진단',
    '상담기록': '04_상담기록',
    '실천자료': '05_실천자료',
    '상담결과': '06_상담결과'
  };

  const target = getOrCreateSubfolder_(
    caseFolder,
    folderMap[payload.fileType] || '06_상담결과'
  );

  const bytes = Utilities.base64Decode(payload.base64);
  const ext = payload.fileName.includes('.')
    ? payload.fileName.substring(payload.fileName.lastIndexOf('.'))
    : '';

  const studentName = sanitizeFileName_(detail.student['학생명'] || '학생');
  const point = String(payload.evaluationPoint || '사전').trim() || '사전';
  const pointText = (payload.fileType === '진단결과' || payload.fileType === '응답내역')
    ? '_' + sanitizeFileName_(point)
    : '';

  const safeName =
    studentName + '_' + payload.fileType + pointText + '_' +
    Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyyMMdd_HHmmss') +
    ext;

  const file = target.createFile(
    Utilities.newBlob(bytes, payload.mimeType, safeName)
  );

  let diagnosisId = '';
  if (payload.fileType === '진단결과' || payload.fileType === '응답내역') {
    const diagnosis = ensureDiagnosisRecord_(payload.caseId, point);
    diagnosisId = diagnosis['진단ID'];

    const patch = {
      '진단ID': diagnosisId,
      '케이스번호': payload.caseId,
      '평가시점': point
    };
    if (payload.fileType === '진단결과') patch['결과PDF_URL'] = file.getUrl();
    if (payload.fileType === '응답내역') patch['응답PDF_URL'] = file.getUrl();

    updateOrAppendByKey_(SHEETS.DIAG, '진단ID', diagnosisId, patch);
  }

  log_('UPLOAD', 'Drive', payload.caseId, payload.fileType + ' 파일 업로드');
  return {
    ok: true,
    fileId: file.getId(),
    fileUrl: file.getUrl(),
    fileName: safeName,
    diagnosisId: diagnosisId,
    evaluationPoint: point
  };
}

function analyzeDiagnosisFiles(payload) {
  validateRequired_(payload, ['caseId', 'resultFileId']);
  assertConsentComplete_(payload.caseId);

  const point = String(payload.evaluationPoint || '사전').trim() || '사전';
  const diagnosis = payload.diagnosisId
    ? getDiagnosisRecordById_(payload.diagnosisId)
    : ensureDiagnosisRecord_(payload.caseId, point);

  const diagnosisId = diagnosis['진단ID'];
  const detail = getCaseDetailRaw_(payload.caseId);
  const resultBlob = DriveApp.getFileById(payload.resultFileId).getBlob();
  const blobs = [{ label: point + ' 진단결과', blob: resultBlob }];

  if (payload.responseFileId) {
    blobs.push({
      label: point + ' 응답내역',
      blob: DriveApp.getFileById(payload.responseFileId).getBlob()
    });
  }

  const extracted = extractDiagnosisWithGemini_(blobs, detail.student);
  const mapped = mapExtractedDiagnosis_(
    payload.caseId,
    detail,
    extracted,
    diagnosisId,
    point
  );

  updateOrAppendByKey_(SHEETS.DIAG, '진단ID', diagnosisId, mapped.sheetRow);

  if (point === '사전') {
    updateCaseStatus(payload.caseId, '결과 확인', '');
  }

  log_(
    'AI_EXTRACT',
    SHEETS.DIAG,
    diagnosisId,
    point + ' PDF 자동 추출 완료'
  );

  return {
    ok: true,
    values: mapped.formValues,
    warnings: mapped.warnings
  };
}

function extractDiagnosisWithGemini_(blobs, student) {

  const apiKey = props.getProperty('GEMINI_API_KEY');
  if (!apiKey) {
    throw new Error('Gemini API 키가 등록되지 않았습니다. 스프레드시트 메뉴의 “AI 영양상담 설정 → Gemini API 키 등록”에서 먼저 등록하세요.');
  }
  const settings = getSettings_();
  const model = String(settings.GEMINI_MODEL || 'gemini-3.1-flash-lite').trim();

  const prompt = [
    '첨부 PDF는 교육부 학교 영양상담용 식생활·생활습관 진단 결과 및 응답내역입니다.',

    '응답내역 PDF가 함께 있으면 상담에 유용한 세부 응답을 responseHighlights 배열에 간단히 정리하세요.',
    '값이 없거나 읽을 수 없으면 빈 문자열을 반환하세요.',
    `선택한 학생 정보: 이름=${student['학생명'] || ''}, 학년=${student['학년'] || ''}, 반=${student['반'] || ''}`,
    '학년·이름·검사일 불일치, 비현실적인 수면시간 등 확인이 필요한 사항은 warnings 배열에 기록하세요.'
  ].join('\n');

  const parts = [{ text: prompt }];
  blobs.forEach(item => {
    parts.push({ text: `[${item.label} PDF]` });
    parts.push({
      inlineData: {
        mimeType: 'application/pdf',
        data: Utilities.base64Encode(item.blob.getBytes())
      }
    });
  });

  const stringProp = description => ({ type: 'string', description });
  const properties = {
    studentName: stringProp('학생 이름'),
    schoolType: stringProp('초등학교/중학교/고등학교'),
    grade: stringProp('학년 숫자'),
    sex: stringProp('성별'),
    age: stringProp('나이'),
    examDate: stringProp('검사일 YYYY-MM-DD'),
    heightCm: stringProp('신장 cm'),
    heightPercentile: stringProp('신장 백분위'),
    weightKg: stringProp('체중 kg'),
    weightPercentile: stringProp('체중 백분위'),
    bmi: stringProp('체질량지수'),
    bmiPercentile: stringProp('체질량지수 백분위'),
    subjectiveHealth: stringProp('주관적 건강상태 판정'),
    bodyImage: stringProp('체형 인식 판정'),
    mealFrequency: stringProp('식사빈도 판정 또는 횟수'),
    regularMealTime: stringProp('규칙적인 식사 시간 판정'),
    eatingSpeed: stringProp('식사속도 판정'),
    mealAmount: stringProp('식사량 판정'),
    totalLevel: stringProp('영양지수 종합 등급'),
    totalScore: stringProp('영양지수 종합 점수'),
    balanceLevel: stringProp('균형 등급'),
    balanceScore: stringProp('균형 점수'),
    moderationLevel: stringProp('절제 등급'),
    moderationScore: stringProp('절제 점수'),
    practiceLevel: stringProp('실천 등급'),
    practiceScore: stringProp('실천 점수'),
    eatingAttitude: stringProp('섭식태도 판정'),
    eatingAttitudeScore: stringProp('섭식태도 점수'),
    allergy: stringProp('식품 알레르기 결과'),
    disease: stringProp('질환 결과'),
    sleepLevel: stringProp('수면습관 판정'),
    sleepDuration: stringProp('수면시간'),
    mentalHealth: stringProp('정신건강 판정'),
    smartphoneUsageLevel: stringProp('스마트폰 사용 습관 판정'),
    weekdaySmartphoneHours: stringProp('주중 하루 평균 스마트폰 사용시간'),
    weekendSmartphoneHours: stringProp('주말 하루 평균 스마트폰 사용시간'),
    smartphoneOverdependence: stringProp('스마트폰 과의존 판정'),
    additionalRequest: stringProp('영양상담 추가 요청사항'),
    warnings: { type: 'array', items: { type: 'string' } },
    responseHighlights: { type: 'array', items: { type: 'string' } }
  };
  const required = Object.keys(properties);

  const requestBody = {
    contents: [{ role: 'user', parts }],
    generationConfig: {
      responseMimeType: 'application/json',
      responseSchema: { type: 'object', properties, required }
    }
  };

  const url = `https://generativelanguage.googleapis.com/v1beta/models/${encodeURIComponent(model)}:generateContent`;
  const response = UrlFetchApp.fetch(url, {
    method: 'post',
    contentType: 'application/json',
    headers: { 'x-goog-api-key': apiKey },
    payload: JSON.stringify(requestBody),
    muteHttpExceptions: true
  });

  const status = response.getResponseCode();
  const body = response.getContentText();
  if (status < 200 || status >= 300) {
    throw new Error(`Gemini PDF 분석 실패 (${status}): ${body.substring(0, 500)}`);
  }

  const parsed = JSON.parse(body);
  const text = (((parsed.candidates || [])[0] || {}).content || {}).parts || [];
  const joined = text.map(p => p.text || '').join('').trim();
  if (!joined) throw new Error('Gemini가 추출 결과를 반환하지 않았습니다.');
  return JSON.parse(joined.replace(/^```json\s*/i, '').replace(/```$/i, '').trim());
}

function mapExtractedDiagnosis_(caseId, detail, data, diagnosisId, evaluationPoint) {
  const student = detail.student || {};
  const existing = getDiagnosisRecordById_(diagnosisId) || {};
  const warnings = Array.isArray(data.warnings) ? data.warnings.slice() : [];

  if (data.studentName && student['학생명'] && normalize_(data.studentName) !== normalize_(student['학생명'])) {
    warnings.push(`PDF 학생명(${data.studentName})과 선택한 학생명(${student['학생명']})이 다릅니다.`);
  }
  if (data.grade && student['학년'] && Number(String(data.grade).replace(/[^0-9]/g, '')) !== Number(student['학년'])) {
    warnings.push(`PDF 학년(${data.grade})과 선택한 학생 학년(${student['학년']})이 다릅니다.`);
  }
  const age = parseNumber_(data.age);
  if (age !== '' && student['학년'] && (age < Number(student['학년']) + 5 || age > Number(student['학년']) + 9)) {
    warnings.push(`학년과 나이(${age}세)의 일치 여부를 확인하세요.`);
  }
  const sleepHours = parseDurationHours_(data.sleepDuration);
  if (sleepHours !== '' && (sleepHours < 4 || sleepHours > 15)) {
    warnings.push(`수면시간(${data.sleepDuration}) 입력값을 확인하세요.`);
  }

  const row = {
    '진단ID': diagnosisId,
    '케이스번호': caseId,
    '검사차수': existing['검사차수'] || getDiagnosisRowsForCase_(caseId).length,
    '평가시점': evaluationPoint || existing['평가시점'] || '사전',
    '학생코드': detail.case['학생코드'],
    '학생명': student['학생명'] || data.studentName || '',
    '학년도': student['학년도'] || '',
    '학년': student['학년'] || parseNumber_(data.grade),
    '반': student['반'] || '',
    '검사일': parseDateOrBlank_(data.examDate),
    '결과PDF_URL': existing['결과PDF_URL'] || '',
    '응답PDF_URL': existing['응답PDF_URL'] || '',
    '성별': data.sex || '',
    '나이': parseNumber_(data.age),
    '신장_cm': parseNumber_(data.heightCm),
    '신장백분위': data.heightPercentile || '',
    '체중_kg': parseNumber_(data.weightKg),
    '체중백분위': data.weightPercentile || '',
    'BMI': parseNumber_(data.bmi),
    'BMI백분위': data.bmiPercentile || '',
    '주관적건강상태': data.subjectiveHealth || '',
    '체형인식': data.bodyImage || '',
    '식사빈도': data.mealFrequency || '',
    '규칙적식사': data.regularMealTime || '',
    '식사속도': data.eatingSpeed || '',
    '식사량': data.mealAmount || '',
    '종합등급': data.totalLevel || '',
    '종합점수': parseNumber_(data.totalScore),
    '균형등급': data.balanceLevel || '',
    '균형점수': parseNumber_(data.balanceScore),
    '절제등급': data.moderationLevel || '',
    '절제점수': parseNumber_(data.moderationScore),
    '실천등급': data.practiceLevel || '',
    '실천점수': parseNumber_(data.practiceScore),

function calculateBmi_(heightCm, weightKg) {
  const h = Number(heightCm);
  const w = Number(weightKg);
  if (!isFinite(h) || !isFinite(w) || h <= 0 || w <= 0) return '';
  return Math.round((w / Math.pow(h / 100, 2)) * 100) / 100;
}

function valueOrBlank_(value) {

function differenceValue_(post, pre, digits) {
  const a = Number(post);
  const b = Number(pre);
  if (!isFinite(a) || !isFinite(b)) return '';
  const factor = Math.pow(10, digits || 0);
  return Math.round((a - b) * factor) / factor;
}

function splitSectionedRecord_(value, labels) {
  const normalized = normalizeSectionedRecord_(value, labels);
  const out = {};
  labels.forEach(label => out[label] = '');
  if (!normalized) return out;

  const pattern = new RegExp(
    '(?:^|\\n\\n)(' + labels.join('|') + ')\\s*[:：]\\s*([\\s\\S]*?)(?=\\n\\n(?:' + labels.join('|') + ')\\s*[:：]|$)',
    'gi'
  );

  let match;
  while ((match = pattern.exec(normalized)) !== null) {
    out[String(match[1]).toUpperCase()] = String(match[2] || '').trim();
  }
  return out;
}

function composeSectionedRecord_(parts, labels) {
  return labels
    .map(label => {
      const value = String(parts[label] || '').trim();
      return value ? label + ': ' + value : '';
    })
    .filter(Boolean)
    .join('\n\n');
}

function normalizeSectionedRecord_(value, labels) {
  let text = String(value || '')
    .replace(/\r\n/g, '\n')
    .replace(/\r/g, '\n')
    .trim();

  if (!text) return '';

  const labelPattern = labels
    .map(label => String(label).replace(/[.*+?^${}()|[\]\\]/g, '\\$&'))
    .join('|');

  const regex = new RegExp(
    '(?:^|\\s+)(' + labelPattern + ')\\s*[:：]\\s*',
    'gi'
  );

  text = text.replace(regex, function (_, label) {
    return '\n\n' + String(label).toUpperCase() + ': ';
  });

  return text
    .replace(/^\s+/, '')
    .replace(/[ \t]+\n/g, '\n')
    .replace(/\n{3,}/g, '\n\n')
    .trim();
}

function normalizeSoap_(value) {
  return normalizeSectionedRecord_(value, ['S', 'O', 'A', 'P']);
}

function normalizePes_(value) {
  return normalizeSectionedRecord_(value, ['P', 'E', 'S']);
}

function getPreviousSessionSummary(caseId) {
  validateRequired_({ caseId }, ['caseId']);

  const detail = getCaseDetailRaw_(caseId);
  const sessions = (detail.sessions || [])
    .slice()
    .sort((a, b) => Number(a['회기'] || 0) - Number(b['회기'] || 0));

  const goals = (detail.goals || []).slice();

  if (!sessions.length) {
    return {
      hasPrevious: false,
      nextSessionNo: 1,
      message: '기존 상담기록이 없습니다. 이번 상담은 1회기입니다.'
    };
  }

  const previousSession = sessions[sessions.length - 1];
  const previousSessionId = String(previousSession['회기ID'] || '');

  let previousGoal = goals
    .filter(goal => String(goal['회기ID'] || '') === previousSessionId)
    .slice(-1)[0];

  if (!previousGoal && goals.length) {
    previousGoal = goals[goals.length - 1];
  }

  previousGoal = previousGoal || {};

  const preparedPlan = (detail.preparations || [])
    .filter(prep => String(prep['기준회기ID'] || '') === previousSessionId)
    .slice(-1)[0] || {};

  return {

    strength: previousSession['학생강점'] || '',
    observation: previousSession['교사관찰'] || '',
    soap: normalizeSoap_(previousSession['SOAP'] || ''),
    pes: normalizePes_(previousSession['PES'] || ''),
    referral: previousSession['연계검토'] || '',
    teacherActivity: [preparedPlan['체험활동'], preparedPlan['행동실험']].filter(Boolean).join('\n\n') || previousSession['교사추천활동'] || '',

    goalText: previousGoal['목표문장'] || '',
    goalStart: formatDate_(previousGoal['시작일']),

    result: previousGoal['결과'] || '',
    studentFeedback: previousGoal['학생소감'] || '',
    difficulty: previousGoal['어려움'] || '',
    supportFactor: previousGoal['도움요인'] || '',
    nextDecision: previousGoal['다음결정'] || '',

  };
}

function formatTeacherActivityPlan_(result) {
  const lines = [];
  const materials = Array.isArray(result.activityMaterials) ? result.activityMaterials : [];
  const steps = Array.isArray(result.activitySteps) ? result.activitySteps : [];
  const questions = Array.isArray(result.activityQuestions) ? result.activityQuestions : [];
  const alternatives = Array.isArray(result.activityAlternatives) ? result.activityAlternatives : [];

  if (result.activityTitle) lines.push('활동명: ' + result.activityTitle);
  if (result.activityPurpose) lines.push('추천 이유·목적: ' + result.activityPurpose);
  if (result.activityMinutes) lines.push('예상 시간: ' + result.activityMinutes);
  if (materials.length) lines.push('준비물: ' + materials.join(', '));

  if (steps.length) {
    lines.push('');
    lines.push('진행 방법');
    steps.forEach((step, index) => lines.push((index + 1) + '. ' + step));
  }

  if (questions.length) {
    lines.push('');
    lines.push('교사 질문 예시');
    questions.forEach(question => lines.push('- ' + question));
  }

  if (result.activityCheck) {
    lines.push('');
    lines.push('학생 산출물·확인: ' + result.activityCheck);
  }

  if (result.activityCaution) {
    lines.push('주의사항: ' + result.activityCaution);
  }

  if (alternatives.length) {
    lines.push('');
    lines.push('대안 활동');
    alternatives.forEach(item => lines.push('- ' + item));
  }

  return lines.join('\n').trim();
}

function splitStoredList_(value) {
  if (Array.isArray(value)) return value.filter(Boolean);
  return String(value || '')
    .split(/\s*[,|·]\s*|\n+/)
    .map(v => v.trim())
    .filter(Boolean);
}

function normalizeListForSheet_(value) {
  if (Array.isArray(value)) return value.map(v => String(v || '').trim()).filter(Boolean).join(' · ');
  return String(value || '').trim();
}

function formatBulletList_(items) {
  const rows = Array.isArray(items) ? items : splitStoredList_(items);
  return rows.map(item => '- ' + String(item || '').trim()).filter(v => v !== '-').join('\n');
}

function inferCounselingArea_(topic, diagnosis) {
  const text = [topic, diagnosis['체형인식'], diagnosis['종합등급'], diagnosis['추가요청사항']]
    .map(v => String(v || '')).join(' ');
  if (/비만|과체중|체중.*증가/.test(text)) return '비만';
  if (/저체중|체중.*부족|마름/.test(text)) return '저체중';
  if (/알레르기/.test(text)) return '알레르기';
  if (/성장|키|신장/.test(text)) return '성장';
  if (/편식|균형|채소|과일|식품군/.test(text)) return '편식';
  return '기타·복합';
}

function inferBarriers_(latestSession, latestGoal, latestCheck) {
  const text = [
    latestSession['학생이말한원인'], latestSession['학생주요호소'],
    latestGoal['어려움'], latestCheck['어려움'], latestCheck['학생소감']
  ].map(v => String(v || '')).join(' ');
  const rows = [];
  if (/모르|지식|이해|정보/.test(text)) rows.push('지식 부족');
  if (/시간|바쁘|늦잠|준비.*시간/.test(text)) rows.push('시간 부족');
  if (/맛|냄새|식감|촉감|소리|감각/.test(text)) rows.push('감각 문제');
  if (/가정|부모|가족|급식|환경|준비되지|구하기/.test(text)) rows.push('환경 문제');
  if (/불안|스트레스|기분|걱정|창피|정서/.test(text)) rows.push('정서 문제');
  if (/싫|의지|귀찮|동기|필요.*모르/.test(text)) rows.push('동기 부족');
  if (/방법|어떻게|기술|고르|준비|실천.*어렵/.test(text)) rows.push('기술 부족');
  return [...new Set(rows)];
}

function buildNextSessionPreparationContext_(caseId) {
  validateRequired_({ caseId }, ['caseId']);
  ensureV46Structure_();

  const detail = getCaseDetailRaw_(caseId);
  const student = detail.student || {};
  const caseRow = detail.case || {};
  const diagnosis = getLatestDiagnosis_(detail) || {};
  const sessions = (detail.sessions || []).slice().sort((a, b) => Number(a['회기'] || 0) - Number(b['회기'] || 0));
  const goals = (detail.goals || []).slice();
  const checks = (detail.goalChecks || []).slice().sort((a, b) => {
    const ad = asDate_(a['점검일']);
    const bd = asDate_(b['점검일']);
    return (ad ? ad.getTime() : 0) - (bd ? bd.getTime() : 0);
  });
  const preparations = (detail.preparations || []).slice().sort((a, b) => {
    const ad = asDate_(a['생성일']);
    const bd = asDate_(b['생성일']);
    return (ad ? ad.getTime() : 0) - (bd ? bd.getTime() : 0);
  });

  const latestSession = sessions.length ? sessions[sessions.length - 1] : {};
  const latestSessionId = String(latestSession['회기ID'] || '');
  let latestGoal = goals.filter(goal => String(goal['회기ID'] || '') === latestSessionId).slice(-1)[0] || {};
  if (!Object.keys(latestGoal).length && goals.length) latestGoal = goals[goals.length - 1];
  const latestGoalId = String(latestGoal['목표ID'] || '');
  const latestCheck = checks.filter(check => String(check['목표ID'] || '') === latestGoalId).slice(-1)[0] || {};
  const nextSessionNo = sessions.length + 1;
  const latestPrep = preparations.filter(prep =>
    Number(prep['준비회기'] || 0) === nextSessionNo ||
    (latestSessionId && String(prep['기준회기ID'] || '') === latestSessionId)
  ).slice(-1)[0] || {};

  let stage = '첫 만남';
  if (sessions.length === 1 && !latestGoal['목표문장']) stage = '탐색';
  else if (sessions.length >= 1 && latestGoal['목표문장'] && !Object.keys(latestCheck).length) stage = '실천';
  else if (sessions.length >= 2 || Object.keys(latestCheck).length) stage = '추적';

  const defaultMinutes = Number(latestSession['상담시간(분)']) || 15;
  const defaultArea = inferCounselingArea_(caseRow['주상담주제'], diagnosis);
  const defaultBarriers = inferBarriers_(latestSession, latestGoal, latestCheck);

  return {
    caseId,
    studentName: student['학생명'] || '',
    gradeClass: formatGradeClass_(student),
    grade: student['학년'] || '',
    caseTopic: caseRow['주상담주제'] || '',
    nextSessionNo: sessions.length + 1,
    defaultArea,
    defaultBarriers,
    defaultMinutes: [10, 15, 20].includes(defaultMinutes) ? defaultMinutes : 15,
    defaultStage: stage,
    latestSession: {
      sessionId: latestSessionId,
      sessionNo: latestSession['회기'] || '',
      sessionDate: formatDate_(latestSession['상담일']),
      mainConcern: latestSession['학생주요호소'] || '',

  };

  const requestBody = {
    contents: [{ role: 'user', parts: [{ text: prompt }] }],
    generationConfig: {
      temperature: 0.25,
      responseMimeType: 'application/json',
      responseSchema: {
        type: 'object',
        properties,
        required: [
          'sessionObjective','questions','counselingTechnique','recommendedMaterials',
          'experienceActivity','behaviorExperiments','nextCheckQuestion','referralCheck',
          'timingPlan','postSessionChecklist','warnings'
        ]
      }
    }
  };

  const url = 'https://generativelanguage.googleapis.com/v1beta/models/' + encodeURIComponent(model) + ':generateContent';
  const response = UrlFetchApp.fetch(url, {
    method: 'post',
    contentType: 'application/json',
    headers: { 'x-goog-api-key': apiKey },
    payload: JSON.stringify(requestBody),
    muteHttpExceptions: true
  });

  const result = JSON.parse(text.replace(/^```json\s*/i, '').replace(/```$/i, '').trim());

  return toClient_({
    ok: true,
    sessionObjective: result.sessionObjective || '',
    questions: formatBulletList_(result.questions || []),
    counselingTechnique: result.counselingTechnique || '',
    recommendedMaterials: formatBulletList_(result.recommendedMaterials || []),
    experienceActivity: result.experienceActivity || '',
    behaviorExperiments: formatBulletList_(result.behaviorExperiments || []),
    nextCheckQuestion: result.nextCheckQuestion || '',
    referralCheck: result.referralCheck || '',
    timingPlan: formatBulletList_(result.timingPlan || []),
    postSessionChecklist: formatBulletList_(result.postSessionChecklist || []),
    warnings: Array.isArray(result.warnings) ? result.warnings : []
  });

function getGoalCheckData(caseId) {
  validateRequired_({ caseId }, ['caseId']);
  ensureV40Structure_();

  const detail = getCaseDetailRaw_(caseId);
  const goals = (detail.goals || []).slice();
  if (!goals.length) {
    throw new Error('점검할 실천목표가 없습니다. 먼저 상담기록에서 SMART 목표를 저장하세요.');
  }

  const goal = goals[goals.length - 1];
  const goalId = String(goal['목표ID'] || '');
  const checks = (detail.goalChecks || [])
    .filter(row => String(row['목표ID'] || '') === goalId)
    .sort((a, b) => {
      const ad = asDate_(a['점검일']);
      const bd = asDate_(b['점검일']);
      return (ad ? ad.getTime() : 0) - (bd ? bd.getTime() : 0);
    });

  return toClient_({
    caseId,
    goalId,
    studentName: detail.student['학생명'] || '',
    gradeClass: formatGradeClass_(detail.student),
    goalText: goal['목표문장'] || '',
    startDate: formatDate_(goal['시작일']),
    endDate: formatDate_(goal['종료일']),
    targetCount: goal['목표횟수'] || '',
    actualCount: goal['실제횟수'] || 0,
    achievementRate: goal['달성률'] || '',
    result: goal['결과'] || '진행 중',
    checkMethod: goal['확인방법'] || '',
    studentFeedback: goal['학생소감'] || '',
    difficulty: goal['어려움'] || '',
    supportFactor: goal['도움요인'] || '',
    nextDecision: goal['다음결정'] || '',
    nextDate: formatDate_(goal['다음상담일']),
    checks: checks.slice().reverse().map(check => ({
      checkId: check['점검ID'] || '',
      checkDate: formatDate_(check['점검일']),
      actualCount: check['실제횟수'] || '',
      achievementRate: check['달성률'] || '',
      result: check['결과'] || '',
      studentFeedback: check['학생소감'] || '',
      difficulty: check['어려움'] || '',
      supportFactor: check['도움요인'] || '',
      nextDecision: check['다음결정'] || '',
      nextDate: formatDate_(check['다음상담일'])
    }))
  });
}

function saveGoalCheck(payload) {
  validateRequired_(payload, ['caseId', 'goalId', 'checkDate', 'actualCount']);
  ensureV40Structure_();

  const goals = getSheetObjects_(SHEETS.GOALS);
  const goal = goals.find(row =>
    String(row['목표ID'] || '') === String(payload.goalId) &&
    String(row['케이스번호'] || '') === String(payload.caseId)
  );

  if (!goal) throw new Error('실천목표를 찾을 수 없습니다.');

  const actualCount = Math.max(0, Number(payload.actualCount) || 0);
  const targetCount = Math.max(0, Number(goal['목표횟수']) || 0);
  const achievementRate = targetCount > 0
    ? Math.round(actualCount / targetCount * 100)
    : '';

  let result = String(payload.result || '').trim();
  if (!result || result === '자동 계산') {
    if (targetCount <= 0) result = '진행 중';
    else if (actualCount >= targetCount) result = '달성';
    else if (actualCount > 0) result = '부분 달성';
    else result = '미달성';
  }

  const checkId = nextId_(
    SHEETS.GOAL_CHECKS,
    'GC-' + payload.caseId,
    '점검ID',
    2
  );

  const checkDate = safeDateForSheet_(payload.checkDate, new Date());
  const nextDecision = String(payload.nextDecision || '').trim();
  let nextDate = payload.nextDate
    ? safeDateForSheet_(payload.nextDate, '')
    : '';
  if (nextDecision === '종결' || nextDecision === '종결 검토') {
    nextDate = '';
  }

  appendObject_(SHEETS.GOAL_CHECKS, {
    '점검ID': checkId,
    '목표ID': payload.goalId,
    '케이스번호': payload.caseId,
    '점검일': checkDate,
    '실제횟수': actualCount,
    '달성률': achievementRate,
    '결과': result,
    '학생소감': payload.studentFeedback || '',
    '어려움': payload.difficulty || '',
    '도움요인': payload.supportFactor || '',
    '다음결정': nextDecision,
    '다음상담일': nextDate,
    '담당자': Session.getActiveUser().getEmail(),
    '비고': payload.note || ''
  });

  updateRowByKey_(SHEETS.GOALS, '목표ID', payload.goalId, {
    '실제횟수': actualCount,
    '달성률': achievementRate,
    '결과': result,
    '학생소감': payload.studentFeedback || '',
    '어려움': payload.difficulty || '',
    '도움요인': payload.supportFactor || '',
    '다음결정': nextDecision,
    '다음상담일': nextDate
  });

  let nextStatus = '실천 중';
  if (nextDecision === '종결') {
    nextStatus = '종결';
  } else if (nextDecision === '종결 검토') {
    nextStatus = '종결 검토';
  } else if (nextDate) {
    nextStatus = '추적상담 예정';
  }

  updateCaseStatus(payload.caseId, nextStatus, nextDate || '');

  syncTasteMindCaseSafe_(payload.caseId, {
    allowCreate: true,
    source: '실천목표 점검 저장'
  });

  log_('CREATE', SHEETS.GOAL_CHECKS, checkId, '실천목표 점검 저장');

  return getStudentCounselingOverview(payload.caseId);
}

    '학생주요호소': payload.mainConcern || '',
    '학생이말한원인': payload.cause || '',
    '학생강점': payload.strength || '',
    '교사관찰': payload.observation || '',
    '변화준비도': numberOrBlank_(payload.readiness),
    'SOAP': soap,
    'PES': pes,

  }

    ok: true,
    caseId: payload.caseId,
    sessionId: sessionId,

function updateCaseStatus(caseId, status, nextDate) {
  if (nextDate) {
    assertDashboardAppointmentCapacity_(
      nextDate,
      caseId,
      ''
    );
  }

  const sheet = getSheet_(SHEETS.CASES);
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const idCol = headers.indexOf('케이스번호');
  const statusCol = headers.indexOf('현재단계');
  const nextCol = headers.indexOf('다음일정');
  const closeCol = headers.indexOf('종결일');
  for (let r = 1; r < data.length; r++) {
    if (String(data[r][idCol]) === String(caseId)) {
      sheet.getRange(r + 1, statusCol + 1).setValue(status);
      if (nextDate !== undefined) sheet.getRange(r + 1, nextCol + 1).setValue(nextDate ? safeDateForSheet_(nextDate, '') : '');
      if (status === '종결' && closeCol >= 0) {
        sheet.getRange(r + 1, closeCol + 1).setValue(new Date());
      }

      const explorerSync = syncTasteMindCaseSafe_(caseId, {
        allowCreate: true,
        source: '상담단계 변경'
      });

      return { ok: true, explorerSync: explorerSync };
    }
  }
  throw new Error('상담 케이스를 찾을 수 없습니다.');
}

/**
 * 나이스 상담관리 업로드 후보 회기를 조회합니다.
 * 상담일자·학년·성별·상담시간(분)만 상담자료에서 가져오고,
 * 나머지 항목은 나이스 예시 서식의 고정값을 사용합니다.
 */
function listNeisUploadRows(filters) {
  filters = filters || {};

  const settings = getSettings_();
  const students = studentMap_();
  const caseMap = {};
  getSheetObjects_(SHEETS.CASES).forEach(row => {
    caseMap[String(row['케이스번호'] || '')] = row;
  });

  const diagnosisSexMap = {};
  getSheetObjects_(SHEETS.DIAG).forEach(row => {
    const caseId = String(row['케이스번호'] || '');
    if (!caseId || !row['성별']) return;

    const existing = diagnosisSexMap[caseId];
    const currentDate = asDate_(row['검사일']);
    const existingDate = existing ? asDate_(existing['검사일']) : null;

    if (!existing || (currentDate && (!existingDate || currentDate > existingDate))) {
      diagnosisSexMap[caseId] = row;
    }
  });

  const startDate = asDate_(filters.startDate);
  const endDate = asDate_(filters.endDate);
  if (endDate) endDate.setHours(23, 59, 59, 999);

  const rows = getSheetObjects_(SHEETS.SESSIONS)
    .map(session => {
      const caseId = String(session['케이스번호'] || '');
      const caseRow = caseMap[caseId] || {};
      const student = students[caseRow['학생코드']] || {};
      const diagnosis = diagnosisSexMap[caseId] || {};
      const sessionDate = asDate_(session['상담일']);
      const grade = student['학년'] || '';
      const sex = student['성별'] || diagnosis['성별'] || '';
      const minutes = numberOrBlank_(session['상담시간(분)']);

      const missing = [];
      if (!sessionDate) missing.push('상담일자');
      if (!grade) missing.push('학년');
      if (!sex) missing.push('성별');
      if (minutes === '') missing.push('상담시간(분)');

      return {
        sessionId: String(session['회기ID'] || ''),
        caseId,
        sessionNo: session['회기'] || '',
        sessionDate: formatDate_(sessionDate),
        neisDate: formatNeisDate_(sessionDate),
        studentName: student['학생명'] || '',
        grade: grade ? String(grade) + '학년' : '',
        sex: normalizeNeisSex_(sex),
        minutes,
        topic: caseRow['주상담주제'] || '',
        approval: session['교사승인'] || '',
        missing,
        ready: missing.length === 0,
        schoolYear: Number(settings.SCHOOL_YEAR || student['학년도'] || new Date().getFullYear())
      };
    })
    .filter(row => row.sessionId)
    .filter(row => {
      const d = asDate_(row.sessionDate);
      if (startDate && (!d || d < startDate)) return false;
      if (endDate && (!d || d > endDate)) return false;
      return true;
    })
    .sort((a, b) => {
      const dateDiff = String(a.sessionDate).localeCompare(String(b.sessionDate));
      if (dateDiff !== 0) return dateDiff;
      return String(a.studentName).localeCompare(String(b.studentName), 'ko');
    });

  return toClient_(rows);
}

function normalizeNeisSex_(value) {
  const text = String(value || '').trim();
  if (!text) return '';
  if (['여', '여자', '여학생', '여성', 'F', 'Female'].includes(text)) return '여';
  if (['남', '남자', '남학생', '남성', 'M', 'Male'].includes(text)) return '남';
  return text;
}

function formatNeisDate_(value) {
  const d = asDate_(value);
  if (!d) return '';
  return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyyMMdd');
}

/**
 * 선택한 상담회기를 나이스 상담관리 업로드용 XLSX 파일로 생성합니다.
 */
function createNeisUploadExcel(sessionIds) {
  if (!Array.isArray(sessionIds) || !sessionIds.length) {
    throw new Error('내보낼 상담회기를 한 건 이상 선택하세요.');
  }

  const selectedSet = new Set(sessionIds.map(String));
  const rows = listNeisUploadRows({})
    .filter(row => selectedSet.has(String(row.sessionId)));

  if (!rows.length) {
    throw new Error('선택한 상담회기를 찾을 수 없습니다.');
  }

  const invalid = rows.filter(row => !row.ready);
  if (invalid.length) {
    const detail = invalid.map(row =>
      `${row.studentName} ${row.sessionNo}회기: ${row.missing.join(', ')} 누락`
    ).join('\n');
    throw new Error('필수값이 누락된 상담회기가 있습니다.\n' + detail);
  }

  const settings = getSettings_();
  const schoolYear = Number(settings.SCHOOL_YEAR || rows[0].schoolYear || new Date().getFullYear());
  const now = new Date();
  const dateStamp = Utilities.formatDate(now, Session.getScriptTimeZone(), 'yyyyMMdd_HHmmss');
  const fileName = `${schoolYear}학년도_영양상담_나이스업로드_${dateStamp}.xlsx`;

  const temp = SpreadsheetApp.create(fileName.replace(/\.xlsx$/i, ''));
  const sheet = temp.getSheets()[0];
  sheet.setName('시트1');

  const values = [NEIS_UPLOAD_HEADERS];
  rows.forEach(row => {
    values.push([
      '일반상담',  // 상담분류
      '일반',      // Wee클래스
      '상담',      // 대분류
      '개인상담',  // 중분류
      '기타',      // 상담구분
      1,           // 상담인원
      schoolYear,  // 학년도
      row.neisDate,
      row.grade,
      row.sex,
      '영양상담',  // 상담제목
      '',          // 상담내용은 입력하지 않음
      0,           // 상담시간(시)
      Number(row.minutes),
      '교사',      // 상담자소속
      '면담'       // 상담매체구분
    ]);
  });

  sheet.getRange(1, 1, values.length, NEIS_UPLOAD_HEADERS.length).setValues(values);
  sheet.setFrozenRows(1);
  sheet.getRange(1, 1, 1, NEIS_UPLOAD_HEADERS.length)
    .setFontWeight('bold')
    .setBackground('#f1f4f8')
    .setHorizontalAlignment('center');
  sheet.getRange(2, 1, Math.max(values.length - 1, 1), NEIS_UPLOAD_HEADERS.length)
    .setVerticalAlignment('middle');
  sheet.getRange(2, 8, Math.max(values.length - 1, 1), 1).setNumberFormat('@');
  sheet.autoResizeColumns(1, NEIS_UPLOAD_HEADERS.length);
  sheet.setColumnWidth(12, 180);

  SpreadsheetApp.flush();

  const exportUrl = 'https://www.googleapis.com/drive/v3/files/' + temp.getId() +
    '/export?mimeType=' + encodeURIComponent('application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');

  const response = UrlFetchApp.fetch(exportUrl, {
    headers: { Authorization: 'Bearer ' + ScriptApp.getOAuthToken() },
    muteHttpExceptions: true
  });

  if (response.getResponseCode() < 200 || response.getResponseCode() >= 300) {
    DriveApp.getFileById(temp.getId()).setTrashed(true);
    throw new Error('엑셀 파일 생성에 실패했습니다: ' + response.getContentText().substring(0, 300));
  }

  const blob = response.getBlob().setName(fileName);
  DriveApp.getFileById(temp.getId()).setTrashed(true);

  return {
    ok: true,
    fileName,
    mimeType: blob.getContentType(),
    base64: Utilities.base64Encode(blob.getBytes()),
    count: rows.length
  };
}

function appendObject_(sheetName, obj) {
  const sheet = getSheet_(sheetName);
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const row = headers.map(h => obj[h] !== undefined ? obj[h] : '');
  sheet.appendRow(row);
  return sheet.getLastRow();
}

function updateOrAppendByKey_(sheetName, keyHeader, keyValue, obj) {
  const sheet = getSheet_(sheetName);
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const keyCol = headers.indexOf(keyHeader);
  if (keyCol < 0) throw new Error(sheetName + ' 시트에서 키 열을 찾을 수 없습니다: ' + keyHeader);
  let rowIndex = -1;
  for (let r = 1; r < data.length; r++) {
    if (String(data[r][keyCol]) === String(keyValue)) { rowIndex = r + 1; break; }
  }
  const row = headers.map((h, i) => obj[h] !== undefined ? obj[h] : (rowIndex > 0 ? data[rowIndex - 1][i] : ''));
  if (rowIndex > 0) sheet.getRange(rowIndex, 1, 1, headers.length).setValues([row]);
  else sheet.appendRow(row);
}

function updateRowByKey_(sheetName, keyHeader, keyValue, patch) {
  const sheet = getSheet_(sheetName);
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const keyCol = headers.indexOf(keyHeader);
  for (let r = 1; r < data.length; r++) {
    if (String(data[r][keyCol]) === String(keyValue)) {
      Object.keys(patch).forEach(key => {
        const col = headers.indexOf(key);
        if (col >= 0) sheet.getRange(r + 1, col + 1).setValue(patch[key]);
      });
      return;
    }
  }
  throw new Error('수정할 행을 찾을 수 없습니다.');
}

function getSheetObjects_(sheetName) {
  const sheet = getSheet_(sheetName);
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return [];
  const headers = data[0];
  return data.slice(1)
    .filter(row => row.some(v => v !== '' && v !== null))
    .map(row => {
      const obj = {};
      headers.forEach((h, i) => obj[h] = row[i]);
      return obj;
    });
}

function nextId_(sheetName, prefix, headerName, digits) {
  const rows = getSheetObjects_(sheetName);

  });
  return prefix + '-' + Utilities.formatString('%0' + digits + 'd', max + 1);
}

function extractDriveId_(url) {

function numberOrBlank_(value) {
  return value === '' || value === null || value === undefined ? '' : Number(value);
}

function parseNumber_(value) {
  if (value === '' || value === null || value === undefined) return '';
  const m = String(value).replace(/,/g, '').match(/-?\d+(?:\.\d+)?/);
  return m ? Number(m[0]) : '';
}

function parseDurationHours_(value) {
  if (!value) return '';
  const text = String(value);
  const hour = text.match(/(\d+(?:\.\d+)?)\s*시간/);
  const minute = text.match(/(\d+)\s*분/);
  if (hour) return Number(hour[1]) + (minute ? Number(minute[1]) / 60 : 0);
  const n = parseNumber_(text);
  return n;
}

function parseDateOrBlank_(value) {
  if (!value) return '';
  const d = new Date(value);
  return isNaN(d.getTime()) ? '' : d;
}

        student['반'],
        student['번호'],
        intake['신청자명'],
        intake['신청내용'],
        intake['비고']
      ].map(value =>
        String(value || '').toLowerCase()
      ).join(' ');

      return {
        caseId:
          caseRow['케이스번호'] || '',
        intakeId:
          caseRow['접수ID'] || '',
        studentCode:
          caseRow['학생코드'] || '',
        studentName:
          student['학생명'] || '',
        gradeClass: [
          student['학년']
            ? student['학년'] + '학년'
            : '',
          student['반']
            ? student['반'] + '반'
            : '',
          student['번호']
            ? student['번호'] + '번'
            : ''
        ].filter(Boolean).join(' '),
        topic:
          caseRow['주상담주제'] || '',
        status:
          caseRow['현재단계'] || '',
        likelyTest:
          isLikelyTestRecord_(
            caseRow,
            student,
            intake
          ),
        searchable: searchable
      };
    })
    .filter(item =>
      item.searchable.includes(keyword)
    )
    .slice(0, 50)
    .map(item => {
      delete item.searchable;
      return item;
    });

  return toClient_(results);
}

/**
 * 선택한 테스트 케이스와 관련 자료를 한 번에 삭제합니다.
 *
 * 삭제 범위
 * - 매니저 관련 시트
 * - 상담접수·상담케이스
 * - 맛마음 탐험소 계정·활동
 * - 보호자동의 PDF·생성문서 Drive 파일
 * - 상담케이스 Drive 폴더
 * - 선택 케이스 삭제 후 다른 케이스가 없는 학생정보
 */
function deleteSelectedTestData(caseIds) {
  if (!Array.isArray(caseIds)) {
    throw new Error(
      '삭제할 상담케이스를 선택하세요.'
    );
  }

  const normalizedIds = Array.from(
    new Set(
      caseIds
        .map(value =>
          String(value || '').trim()
        )
        .filter(Boolean)
    )
  );

  if (!normalizedIds.length) {
    throw new Error(
      '삭제할 상담케이스를 한 개 이상 선택하세요.'
    );
  }

  if (normalizedIds.length > 50) {
    throw new Error(
      '한 번에 최대 50개까지만 삭제할 수 있습니다.'
    );
  }

  const lock =
    LockService.getScriptLock();

  lock.waitLock(30000);

  try {
    const selectedSet =
      new Set(normalizedIds);

    const allCases =
      getSheetObjects_(SHEETS.CASES);

    const selectedCases =
      allCases.filter(row =>
        selectedSet.has(
          String(row['케이스번호'] || '')
        )
      );

    if (!selectedCases.length) {
      throw new Error(
        '선택한 상담케이스를 찾지 못했습니다.'
      );
    }

    const foundCaseIds =
      new Set(
        selectedCases.map(row =>
          String(row['케이스번호'] || '')
        )
      );

    const missingCaseIds =
      normalizedIds.filter(id =>
        !foundCaseIds.has(id)
      );

    const intakeIds =
      new Set(
        selectedCases
          .map(row =>
            String(row['접수ID'] || '')
          )
          .filter(Boolean)
      );

    const studentCodes =
      new Set(
        selectedCases
          .map(row =>
            String(row['학생코드'] || '')
          )
          .filter(Boolean)
      );

    const studentsByCode = {};

    getSheetObjects_(SHEETS.STUDENTS)
      .forEach(row => {
        studentsByCode[
          String(row['학생코드'] || '')
        ] = row;
      });

    /*
     * Drive 파일·폴더는 시트 행을 삭제하기 전에 수집합니다.
     */
    const driveItemIds =
      collectTestCleanupDriveItemIds_(
        selectedSet,
        selectedCases
      );

    /*
     * 탐험소 자료 역시 매니저 케이스가 남아 있을 때
     * 정확한 케이스번호로 먼저 삭제합니다.
     */
    const explorerResult =
      deleteSelectedExplorerTestData_(
        selectedSet
      );

    const deleted = {};

    TEST_CLEANUP_MANAGER_TARGETS
      .forEach(target => {
        deleted[target.sheetName] =
          deleteRowsByPredicate_(
            getSheet_(target.sheetName),
            row =>
              selectedSet.has(
                String(
                  row[target.keyHeader] || ''
                )
              )
          );
      });

    deleted[SHEETS.LOGS] =
      deleteRowsByPredicate_(
        getSheet_(SHEETS.LOGS),
        row => {
          const targetId =
            String(row['대상ID'] || '');

          const changeText =
            String(row['변경내용'] || '');

          return (
            selectedSet.has(targetId) ||
            normalizedIds.some(caseId =>
              changeText.includes(caseId)
            )
          );
        }
      );

    deleted[SHEETS.INTAKE] =
      deleteRowsByPredicate_(
        getSheet_(SHEETS.INTAKE),
        row =>
          intakeIds.has(
            String(row['접수ID'] || '')
          )
      );

    deleted[SHEETS.CASES] =
      deleteRowsByPredicate_(
        getSheet_(SHEETS.CASES),
        row =>
          selectedSet.has(
            String(row['케이스번호'] || '')
          )
      );

    /*
     * 선택 케이스 삭제 후 다른 상담케이스가 남지 않은
     * 학생만 학생정보에서 자동 삭제합니다.
     */
    const remainingCases =
      getSheetObjects_(SHEETS.CASES);

    const removableStudentCodes =
      new Set(
        Array.from(studentCodes)
          .filter(studentCode =>
            !remainingCases.some(row =>
              String(row['학생코드'] || '') ===
              studentCode
            )
          )
      );

    deleted[SHEETS.STUDENTS] =
      deleteRowsByPredicate_(
        getSheet_(SHEETS.STUDENTS),
        row =>
          removableStudentCodes.has(
            String(row['학생코드'] || '')
          )
      );

    Object.keys(explorerResult.deleted)
      .forEach(sheetName => {
        deleted[
          '탐험소·' + sheetName
        ] =
          explorerResult.deleted[sheetName];
      });

    const driveResult =
      trashTestCleanupDriveItems_(
        driveItemIds
      );

    const deletedCases =
      selectedCases.map(row => {
        const student =
          studentsByCode[
            String(row['학생코드'] || '')
          ] || {};

        return {
          caseId:
            row['케이스번호'] || '',
          studentName:
            student['학생명'] || '',
          topic:
            row['주상담주제'] || ''
        };
      });

    return toClient_({
      ok: true,
      deletedCases: deletedCases,
      deleted: deleted,
      driveTrashedCount:
        driveResult.trashedCount,
      missingCaseIds: missingCaseIds,
      warnings: []
        .concat(
          explorerResult.warnings || []
        )
        .concat(
          driveResult.warnings || []
        )
    });
  } finally {
    lock.releaseLock();
  }
}

/**
 * 선택한 케이스와 연결된 Drive 파일·폴더 ID를 수집합니다.
 */
function collectTestCleanupDriveItemIds_(
  selectedSet,
  selectedCases
) {
  const ids = new Set();

  selectedCases.forEach(caseRow => {
    addDriveIdFromValue_(
      ids,
      caseRow['Drive폴더URL']
    );
  });

  TEST_CLEANUP_MANAGER_TARGETS
    .forEach(target => {
      const rows =
        getSheetObjects_(
          target.sheetName
        ).filter(row =>
          selectedSet.has(
            String(
              row[target.keyHeader] || ''
            )
          )
        );

      rows.forEach(row => {
        Object.keys(row).forEach(header => {
          const value = row[header];

          if (
            /URL|링크|파일|문서/i
              .test(String(header || ''))
          ) {
            addDriveIdFromValue_(
              ids,
              value
            );
          }
        });
      });
    });

  return Array.from(ids);
}

function addDriveIdFromValue_(set, value) {
  const text =
    String(value || '').trim();

  if (
    !text ||
    !/^https?:\/\//i.test(text) ||
    !/(drive|docs)\.google\.com/i
      .test(text)
  ) {
    return;
  }

  const id = extractDriveId_(text);

  if (id) set.add(id);
}

/**
 * DriveApp에서는 완전 영구 삭제 대신 휴지통 이동을 지원합니다.
 * 상담 폴더를 휴지통으로 보내면 그 안의 파일도 함께 정리됩니다.
 */
function trashTestCleanupDriveItems_(
  itemIds
) {
  let trashedCount = 0;
  const warnings = [];

  Array.from(
    new Set(itemIds || [])
  ).forEach(itemId => {
    let handled = false;

    try {
      const file =
        DriveApp.getFileById(itemId);

      file.setTrashed(true);
      trashedCount += 1;
      handled = true;
    } catch (fileError) {
      // 파일이 아니라 폴더일 수 있으므로 아래에서 다시 시도합니다.
    }

    if (handled) return;

    try {
      const folder =
        DriveApp.getFolderById(itemId);

      folder.setTrashed(true);
      trashedCount += 1;
      handled = true;
    } catch (folderError) {
      warnings.push(
        'Drive 항목을 휴지통으로 이동하지 못했습니다: ' +
        itemId
      );
    }
  });

  return {
    trashedCount: trashedCount,
    warnings: warnings
  };
}

/**
 * 맛마음 탐험소에서 선택한 케이스번호와 연결된
 * 학생계정·활동을 삭제합니다.
 */
function deleteSelectedExplorerTestData_(
  selectedSet
) {
  const deleted = {};
  const warnings = [];

  try {
    const config =
      getTasteMindConfig_();

    const ss =
      SpreadsheetApp.openById(
        config.spreadsheetId
      );

    const accountSheet =
      ss.getSheetByName(
        TASTE_MIND_DEFAULTS.ACCOUNT_SHEET
      );

    if (!accountSheet) {
      return {
        deleted: deleted,
        warnings: [
          '탐험소 학생계정 시트를 찾지 못했습니다.'
        ]
      };
    }

    const accountRows =
      getExternalSheetObjects_(
        ss,
        TASTE_MIND_DEFAULTS.ACCOUNT_SHEET
      ).filter(row =>
        selectedSet.has(
          String(row['케이스번호'] || '')
        )
      );

    const accountIds =
      new Set(
        accountRows
          .map(row =>
            String(row['계정ID'] || '')
          )
          .filter(Boolean)
      );

    TEST_CLEANUP_EXPLORER_SHEETS
      .forEach(sheetName => {
        const sheet =
          ss.getSheetByName(sheetName);

        if (!sheet) {
          deleted[sheetName] = 0;
          return;
        }

        deleted[sheetName] =
          deleteRowsByPredicate_(
            sheet,
            row => {
              const rowCaseId =
                String(
                  row['케이스번호'] || ''
                );

              const rowAccountId =
                String(
                  row['계정ID'] || ''
                );

              return (
                selectedSet.has(rowCaseId) ||
                (
                  rowAccountId &&
                  accountIds.has(rowAccountId)
                )
              );
            }
          );
      });
  } catch (error) {
    warnings.push(
      '맛마음 탐험소 자료 삭제 실패: ' +
      (
        error &&
        error.message
          ? error.message
          : String(error)
      )
    );
  }

  return {
    deleted: deleted,
    warnings: warnings
  };
}

function deleteRowsByPredicate_(
  sheet,
  predicate
) {
  const data =
    sheet.getDataRange().getValues();

  if (data.length < 2) return 0;

  const headers =
    data[0].map(value =>
      String(value || '').trim()
    );

  const deleteRows = [];

  for (
    let rowIndex = 1;
    rowIndex < data.length;
    rowIndex += 1
  ) {
    const object = {};

    headers.forEach(
      (header, column) => {
        object[header] =
          data[rowIndex][column];
      }
    );

    if (predicate(object)) {
      deleteRows.push(rowIndex + 1);
    }
  }

  deleteRows
    .sort((a, b) => b - a)
    .forEach(rowNumber =>
      sheet.deleteRow(rowNumber)
    );

  return deleteRows.length;
}

function isLikelyTestRecord_(
  caseRow,
  student,
  intake
) {
  const text = [
    student['학생명'],
    caseRow['주상담주제'],
    caseRow['비고'],
    intake['신청자명'],
    intake['신청내용'],
    intake['비고']
  ].map(value =>
    String(value || '')
  ).join(' ');

  return /테스트|시험|샘플|연습|가상|demo|test/i
    .test(text);
}


/**
 * 연락처 값이 날짜 객체나 1970 ISO 문자열로 잘못 들어온 경우
 * 화면에 날짜를 노출하지 않고 재입력이 필요하도록 처리합니다.
 */
function normalizeContactValue_(value) {
  if (value === '' || value === null || value === undefined) return '';

  if (Object.prototype.toString.call(value) === '[object Date]') {
    return '';
  }

  let text = String(value).trim();

  if (
    /^1970-01-01T/i.test(text) ||
    /^Thu Jan 01 1970/i.test(text)
  ) {
    return '';
  }

  let digits = text.replace(/\D/g, '');

  // 앞자리 0이 숫자 변환으로 사라진 일부 휴대전화 번호 복구
  if (digits.length === 10 && digits.indexOf('10') === 0) {
    digits = '0' + digits;
  }

  if (/^01[016789]\d{7,8}$/.test(digits)) {
    if (digits.length === 10) {
      return digits.slice(0, 3) + '-' +
        digits.slice(3, 6) + '-' +
        digits.slice(6);
    }

    return digits.slice(0, 3) + '-' +
      digits.slice(3, 7) + '-' +
      digits.slice(7);
  }

  return text;
}

/**
 * 상담접수·보호자동의 연락처 열을 텍스트 형식으로 바꾸고,
 * 날짜로 잘못 저장된 연락처를 비운 뒤 재입력 위치를 표시합니다.
 *
 * 실행 후 빨간색으로 표시된 셀에 실제 연락처를 다시 입력하세요.
 */
function repairGuardianContactColumns() {
  const targets = [
    {
      sheetName: SHEETS.INTAKE,
      headerName: '연락처',
      idHeader: '접수ID'
    },
    {
      sheetName: SHEETS.CONSENTS,
      headerName: '보호자연락처',
      idHeader: '케이스번호'
    }
  ];

  const suspicious = [];
  let normalizedCount = 0;

  targets.forEach(target => {
    const sheet = getSheet_(target.sheetName);
    const values = sheet.getDataRange().getValues();

    if (!values.length) return;

    const headers = values[0].map(value => String(value || '').trim());
    const contactIndex = headers.indexOf(target.headerName);
    const idIndex = headers.indexOf(target.idHeader);

    if (contactIndex < 0) return;

    sheet.getRange(
      2,
      contactIndex + 1,
      Math.max(sheet.getMaxRows() - 1, 1),
      1
    ).setNumberFormat('@');

    for (let rowIndex = 1; rowIndex < values.length; rowIndex += 1) {
      const raw = values[rowIndex][contactIndex];
      const cell = sheet.getRange(rowIndex + 1, contactIndex + 1);
      const id = idIndex >= 0
        ? String(values[rowIndex][idIndex] || '')
        : String(rowIndex + 1);

      const isDate =
        Object.prototype.toString.call(raw) === '[object Date]';

      const isEpochText =
        /^1970-01-01T/i.test(String(raw || ''));

      if (isDate || isEpochText) {
        cell
          .setNumberFormat('@')
          .setValue('')
          .setBackground('#fff0f0')
          .setNote(
            '연락처가 날짜로 잘못 저장되어 비웠습니다. 실제 보호자 연락처를 텍스트로 다시 입력하세요.'
          );

        suspicious.push(
          target.sheetName + ' · ' + id
        );
        continue;
      }

      const normalized = normalizeContactValue_(raw);

      if (normalized !== String(raw || '')) {
        cell
          .setNumberFormat('@')
          .setValue(normalized)
          .setBackground(null)
          .setNote('');

        normalizedCount += 1;
      }
    }
  });

  SpreadsheetApp.getUi().alert(
    '보호자 연락처 열 정리 완료',
    [
      '정상화한 연락처: ' + normalizedCount + '건',
      '날짜로 잘못 저장되어 재입력이 필요한 셀: ' +
        suspicious.length + '건',
      '',
      suspicious.length
        ? '재입력 대상:\n- ' + suspicious.join('\n- ')
        : '재입력이 필요한 연락처가 없습니다.',
      '',
      '빨간색 셀에는 실제 보호자 휴대전화 번호를 다시 입력하세요.'
    ].join('\n'),
    SpreadsheetApp.getUi().ButtonSet.OK
  );

  return {
    ok: true,
    normalizedCount: normalizedCount,
    suspicious: suspicious
  };
}


function toClient_(value) {
  if (value === null || value === undefined) return value;

  if (Object.prototype.toString.call(value) === '[object Date]') {
    if (isNaN(value.getTime())) return '';
    return Utilities.formatDate(value, Session.getScriptTimeZone(), "yyyy-MM-dd'T'HH:mm:ss");
  }

  if (Array.isArray(value)) return value.map(toClient_);

  if (typeof value === 'number') return isFinite(value) ? value : '';

  if (typeof value === 'object') {
    const out = {};
    Object.keys(value).forEach(key => out[key] = toClient_(value[key]));
    return out;
  }

  return value;
}

function safeDateForSheet_(value, fallback) {
  const d = asDate_(value);
  if (d) return d;
  const fb = asDate_(fallback);
  return fb || '';
}

function asDate_(value) {
  if (value === '' || value === null || value === undefined) return null;
  if (Object.prototype.toString.call(value) === '[object Date]') {
    // Apps Script는 Invalid Date를 클라이언트로 반환하거나 Utilities.formatDate에 넘기면
    // 'Cannot return an invalid date' 오류를 발생시킵니다.
    return isNaN(value.getTime()) ? null : value;
  }
  const d = new Date(value);
  return isNaN(d.getTime()) ? null : d;
}

function stripTime_(date) {
  return new Date(date.getFullYear(), date.getMonth(), date.getDate());
}

function formatDate_(value) {
  const d = asDate_(value);
  return d ? Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd') : '';
}

function formatDateTime_(value) {
  const d = asDate_(value);
  return d ? Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm') : '';
}

function formatGradeClass_(student) {
  const grade = student['학년'] || '';
  const classNo = student['반'] || '';
  return grade && classNo ? `${grade}학년 ${classNo}반` : '';
}

function normalize_(value) {
  return String(value || '').replace(/\s+/g, '').toLowerCase();
}

function sanitizeFileName_(value) {
  return String(value || '').replace(/[\\/:*?"<>|]/g, '_').trim();
}

function appendText_(base, extra) {
  return [base, extra].filter(Boolean).join('\n');
}
